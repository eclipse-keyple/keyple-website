37 Tag::Tag(const std::vector<uint8_t>& binary, int offset)
38 {
39  if (offset >= static_cast<int>(binary.size()))
40  throw std::invalid_argument("offset out of bound");
41 
42  /* the 2 first bits (b7b6) of the first byte defines the class */
43  mTagClass = static_cast<TagClass>((binary[offset] & 0xC0) >> 6);
44 
45  /* the type bit is the third bit (b5) */
46  if ((binary[offset] & 0x20) == 0x20) {
47  mTagType = TagType::CONSTRUCTED;
48  } else {
49  mTagType = TagType::PRIMITIVE;
50  }
51 
52  /* */
53  int number = binary[offset] & 0x1F;
54  if (number == 0x1F) {
55  /* two-byte tag */
56  number = binary[offset + 1];
57  mTagSize = 2;
58  } else {
59  /* one-byte tag */
60  mTagSize = 1;
61  }
62 
63  mTagNumber = number;
64 }
