39 SamSelector::SamSelector(SamSelectorBuilder* builder)
40 : SeSelector(builder), mTargetSamRevision(builder->mSamRevision), mUnlockData(builder->mUnlockData)
41 {
42  std::string atrRegex;
43  std::string snRegex;
44 
45  /* Check if serialNumber is defined */
46  if (builder->mSerialNumber == "")
47  /* Match all serial numbers */
48  snRegex = ".{8}";
49  else
50  /* Match the provided serial number (could be a regex substring) */
51  snRegex = builder->mSerialNumber;
52 
53  /*
54  * Build the final Atr regex according to the SAM subtype and serial number if any.
55  *
56  * The header is starting with 3B, its total length is 4 or 6 bytes (8 or 10 hex digits)
57  */
58  if (mTargetSamRevision == SamRevision::C1 ||
59  mTargetSamRevision == SamRevision::S1D ||
60  mTargetSamRevision == SamRevision::S1E)
61  atrRegex = "3B(.{6}|.{10})805A..80" +
62  mTargetSamRevision.getApplicationTypeMask() +
63  "20.{4}" +
64  snRegex +
65  "829000";
66  else if (mTargetSamRevision == SamRevision::AUTO)
67  /* Match any ATR */
68  atrRegex = ".*";
69  else
70  throw IllegalArgumentException("Unknown SAM subtype.");
71 
72  getAtrFilter()->setAtrRegex(atrRegex);
73 }
