81 std::shared_ptr<SelectionStatus> AbstractLocalReader::openLogicalChannel(
82  std::shared_ptr<SeSelector> seSelector)
83 {
84  const std::vector<uint8_t>& atr = getATR();
85  bool selectionHasMatched = true;
86  std::shared_ptr<SelectionStatus> selectionStatus;
87 
88  /* Perform ATR filtering if requested */
89  if (seSelector->getAtrFilter() != nullptr) {
90  if (atr.empty())
91  throw KeypleReaderIOException("Didn't get an ATR from the SE");
92 
93  mLogger->trace("[%] openLogicalChannel => ATR = %\n", getName(), atr);
94 
95  if (!seSelector->getAtrFilter()->atrMatches(atr)) {
96  mLogger->info("[%] openLogicalChannel => ATR didn't match. SELECTOR = %, ATR = %\n",
97  getName(),
98  seSelector,
99  atr);
100 
101  selectionHasMatched = false;
102  }
103  }
104 
105  /*
106  * Perform application selection if requested and if ATR filtering matched or was not requested
107  */
108  if (selectionHasMatched && seSelector->getAidSelector() != nullptr) {
109  std::shared_ptr<ApduResponse> fciResponse;
110 
111  auto smartSelectionReader = dynamic_cast<SmartSelectionReader *>(this);
112 
113  if (smartSelectionReader) {
114  fciResponse = smartSelectionReader->openChannelForAid(
115  *(seSelector->getAidSelector().get()));
116  } else {
117  fciResponse = processExplicitAidSelection(*(seSelector->getAidSelector().get()));
118  }
119 
120  if (fciResponse->isSuccessful() && fciResponse->getDataOut().size() == 0) {
121  /*
122  * The selection didn't provide data (e.g. OMAPI), we get the FCI using a Get Data
123  * command.
124  *
125  * The AID selector is provided to handle successful status word in the Get Data
126  * command.
127  */
128  fciResponse = recoverSelectionFciData(*(seSelector->getAidSelector().get()));
129  }
130 
131  /*
132  * The ATR filtering matched or was not requested. The selection status is determined by the
133  * answer to the select application command.
134  */
135  selectionStatus = std::make_shared<SelectionStatus>(std::make_shared<AnswerToReset>(atr),
136  fciResponse,
137  fciResponse->isSuccessful());
138  } else {
139  /*
140  * The ATR filtering didn't match or no AidSelector was provided. The selection status is
141  * determined by the ATR filtering.
142  */
143  std::vector<uint8_t> dummy;
144  selectionStatus = std::make_shared<SelectionStatus>(
145  std::make_shared<AnswerToReset>(atr),
146  std::make_shared<ApduResponse>(dummy, nullptr),
147  selectionHasMatched);
148  }
149 
150  return selectionStatus;
151 }

212 std::vector<std::shared_ptr<SeResponse>> AbstractLocalReader::processSeRequests(
213  const std::vector<std::shared_ptr<SeRequest>>& seRequests,
214  const MultiSeRequestProcessing& multiSeRequestProcessing,
215  const ChannelControl& channelControl)
216 {
217  std::vector<bool> requestMatchesProtocol(seRequests.size());
218  int requestIndex = 0, lastRequestIndex;
219 
220  /* Determine which requests are matching the current ATR */
221  mLogger->trace("processSeRequestSet - determining which requests are matching the ATR\n");
222 
223  for (auto request : seRequests) {
224  std::shared_ptr<SeSelector> seSelector = request->getSeSelector();
225  if (seSelector != nullptr) {
226  requestMatchesProtocol[requestIndex] =
227  protocolFlagMatches(request->getSeSelector()->getSeProtocol());
228  } else {
229  requestMatchesProtocol[requestIndex] = true;
230  }
231 
232  requestIndex++;
233  }
234 
235  /*
236  * we have now an array of booleans saying whether the corresponding request and the current SE
237  * match or not
238  */
239  lastRequestIndex = requestIndex;
240  requestIndex = 0;
241 
242  /*
243  * The current request list is possibly made of several APDU command lists.
244  *
245  * If the requestMatchesProtocol is true we process the SeRequest.
246  *
247  * If the requestMatchesProtocol is false we skip to the next SeRequest.
248  *
249  * If keepChannelOpen is false, we close the physical channel for the last request.
250  */
251  mLogger->debug("processSeRequestSet - processing requests set\n");
252 
253  std::vector<std::shared_ptr<SeResponse>> responses;
254  bool stopProcess = false;
255 
256  for (auto request : seRequests) {
257  if (!stopProcess) {
258  if (requestMatchesProtocol[requestIndex]) {
259  mLogger->debug("[%] processSeRequests => transmit %\n", getName(), request);
260 
261  std::shared_ptr<SeResponse> response = nullptr;
262 
263  try {
264  response = processSeRequestLogical(request);
265  } catch (KeypleReaderIOException& ex) {
266  /*
267  * The process has been interrupted. We launch a KeypleReaderException with
268  * the responses collected so far.
269  */
270 
271  /* Add the latest (and partial) SeResponse to the current list */
272  responses.push_back(ex.getSeResponse());
273 
274  /* Build a SeResponseSet with the available data. */
275  ex.setSeResponses(responses);
276  mLogger->debug("[%] processSeRequests => transmit : process interrupted, " \
277  "collect previous responses %\n",
278  getName(),
279  responses);
280  throw ex;
281  }
282 
283  responses.push_back(response);
284  mLogger->debug("[%] processSeRequests => receive %\n", getName(), response);
285  } else {
286  /*
287  * In case the protocolFlag of a SeRequest doesn't match the reader status, a null
288  * SeResponse is added to the SeResponseSet.
289  */
290  responses.push_back(nullptr);
291  }
292 
293  if (multiSeRequestProcessing == MultiSeRequestProcessing::PROCESS_ALL) {
294  /*
295  * Multi SeRequest case: just close the logical channel and go on with the next
296  * selection.
297  */
298  closeLogicalChannel();
299  } else {
300  if (mLogicalChannelIsOpen) {
301  /* The current PO matches the selection case, we stop here */
302  stopProcess = true;
303  }
304  }
305 
306  requestIndex++;
307  if (lastRequestIndex == requestIndex && channelControl != ChannelControl::KEEP_OPEN) {
308  /* Close logical channel unconditionally */
309  closeLogicalChannel();
310 
311  auto observableReader = dynamic_cast<ObservableReader *>(this);
312 
313  /* Not observable or no observers ? */
314  if (!observableReader || observableReader->countObservers() == 0)
315  /* Close immediately the physical channel */
316  closePhysicalChannel();
317  else
318  /* Manage the end of communication in observed mode */
319  terminateSeCommunication();
320  }
321  }
322  }
323 
324  return responses;
325 }

327 std::shared_ptr<SeResponse> AbstractLocalReader::processSeRequest(
328  const std::shared_ptr<SeRequest> seRequest,
329  const ChannelControl& channelControl)
330 {
331  std::shared_ptr<SeResponse> seResponse = nullptr;
332 
333  /* The SeRequest may be null when we just need to close the physical channel */
334  if (seRequest)
335  seResponse = processSeRequestLogical(seRequest);
336 
337  if (channelControl != ChannelControl::KEEP_OPEN) {
338  /* Close logical channel unconditionally */
339  closeLogicalChannel();
340 
341  auto observableReader = dynamic_cast<ObservableReader *>(this);
342 
343  if (observableReader && observableReader->countObservers() == 0)
344  /*
345  * Not observable/observed: close immediately the physical channel
346  * if requested.
347  */
348  closePhysicalChannel();
349 
350  auto abstractReader = dynamic_cast<AbstractObservableLocalReader *>(this);
351 
352  if (abstractReader)
353  /* Request the removal sequence when the reader is monitored by a thread */
354  abstractReader->terminateSeCommunication();
355  }
356 
357  return seResponse;
358 }

360 std::shared_ptr<SeResponse> AbstractLocalReader::processSeRequestLogical(
361  std::shared_ptr<SeRequest> seRequest)
362 {
363  bool previouslyOpen = true;
364  std::shared_ptr<SelectionStatus> selectionStatus = nullptr;
365 
366  std::vector<std::shared_ptr<ApduResponse>> apduResponses;
367 
368  mLogger->debug("[%] processSeRequest => Logical channel open = %\n",
369  getName(),
370  isLogicalChannelOpen());
371 
372  /*
373  * Unless the selector is null, we try to open a logical channel; if the
374  * channel was open and the PO is still matching we won't redo the selection
375  * and just use the current selection status
376  */
377  if (seRequest->getSeSelector() != nullptr) {
378  /* check if AID changed if the channel is already open */
379  if (isLogicalChannelOpen() &&
380  seRequest->getSeSelector()->getAidSelector() != nullptr) {
381  /*
382  * AID comparison hack: we check here if the initial selection AID
383  * matches the beginning of the AID provided in the SeRequest
384  * (coming from FCI data and supposed to be longer than the
385  * selection AID).
386  *
387  * The current AID (selector) length must be at least equal or
388  * greater than the selection AID. All bytes of the selection AID
389  * must match the beginning of the current AID.
390  */
391  if (mAidCurrentlySelected.empty())
392  throw IllegalStateException(
393  "AID currently selected shouldn't be null.");
394 
395  if (seRequest->getSeSelector()
396  ->getAidSelector()
397  ->getFileOccurrence() ==
398  SeSelector::AidSelector::FileOccurrence::NEXT) {
399  mLogger->trace("[%] processSeRequest => The current selection"
400  " is a next selection, close the logical "
401  "channel\n", getName());
402 
403  /*
404  * Close the channel (will reset the current selection status)
405  */
406  closeLogicalChannel();
407 
408  } else if (!startsWith(mAidCurrentlySelected,
409  seRequest->getSeSelector()
410  ->getAidSelector()
411  ->getAidToSelect())) {
412  /*
413  * The AID changed (longer or different), close the logical
414  * channel.
415  */
416  mLogger->debug(
417  "[%] processSeRequest => The AID changed, "
418  "close the logical channel. AID = %, "
419  "EXPECTEDAID = %", getName(), mAidCurrentlySelected,
420  seRequest->getSeSelector());
421 
422  /*
423  * Close the channel (will reset the current selection status)
424  */
425  closeLogicalChannel();
426  }
427 
428  /*
429  * Keep the current selection status (may be null if the current PO
430  * didn't match)
431  */
432  selectionStatus = mCurrentSelectionStatus;
433  }
434 
435  /* open the channel and do the selection if needed */
436  if (!isLogicalChannelOpen()) {
437  previouslyOpen = false;
438 
439  try {
440  selectionStatus =
441  openLogicalChannelAndSelect(seRequest->getSeSelector());
442  mLogger->trace("[%] processSeRequest => Logical channel "
443  "opening success\n", getName());
444  } catch (const IllegalArgumentException& e) {
445  (void)e;
446  mLogger->debug("[%] processSeRequest => Logical channel "
447  "opening failure\n", getName());
448  closeLogicalChannel();
449 
450  /*
451  * Return a null SeResponse when the opening of the logical
452  * channel failed
453  */
454  return nullptr;
455  }
456 
457  if (selectionStatus->hasMatched()) {
458  /*
459  * The selection process succeeded, the logical channel is open
460  */
461  mLogicalChannelIsOpen = true;
462 
463  if (selectionStatus->getFci()->isSuccessful()) {
464  /* The selection AID based was successful, keep the aid */
465  mAidCurrentlySelected = seRequest->getSeSelector()
466  ->getAidSelector()
467  ->getAidToSelect();
468  }
469 
470  mCurrentSelectionStatus = selectionStatus;
471  } else {
472  /* The selection process failed, close the logical channel */
473  closeLogicalChannel();
474  }
475  }
476  } else {
477  /*
478  * Selector is null, we expect that the logical channel was previously
479  * opened
480  */
481  if (!isLogicalChannelOpen())
482  throw IllegalStateException(
483  "[" + getName() + "]" +
484  "processSeRequest => No logical channel opened!");
485  }
486 
487  /* process request if not empty */
488  if (seRequest->getApduRequests().size() > 0) {
489  for (auto apduRequest : seRequest->getApduRequests()) {
490  try {
491  apduResponses.push_back(processApduRequest(apduRequest));
492  } catch (KeypleReaderIOException& ex) {
493  /*
494  * The process has been interrupted. We close the logical
495  * channel and launch a KeypleReaderException with the Apdu
496  * responses collected so far.
497  */
498  mLogger->debug("The process has been interrupted, collect Apdu"
499  "responses collected so far\n");
500  closeLogicalAndPhysicalChannels();
501  ex.setSeResponse(std::make_shared<SeResponse>(
502  false, previouslyOpen, selectionStatus, apduResponses));
503  throw ex;
504  }
505  }
506  }
507 
508  return std::make_shared<SeResponse>(isLogicalChannelOpen(), previouslyOpen,
509  selectionStatus, apduResponses);
510 }

512 bool AbstractLocalReader::startsWith(const std::vector<uint8_t>& source,
513  const std::vector<uint8_t>& match)
514 {

516  return false;
517

520  return false;
521  }

526 std::shared_ptr<ApduResponse> AbstractLocalReader::processApduRequest(
527  std::shared_ptr<ApduRequest> apduRequest)
528 {
529  std::shared_ptr<ApduResponse> apduResponse;
530  long long timeStamp = System::nanoTime();
531  long long elapsed10ms = (timeStamp - mBefore) / 100000;
532  mBefore = timeStamp;
533 
534  mLogger->debug("[%] processApduRequest => %, elapsed % ms\n", getName(),
535  apduRequest, elapsed10ms / 10);
536 
537  const std::vector<uint8_t>& buffer = apduRequest->getBytes();
538  std::vector<uint8_t> rapdu;
539  try {
540  rapdu = transmitApdu(buffer);
541  } catch (const KeypleReaderIOException& e) {
542  (void)e;
543  throw;
544  }
545 
546  apduResponse = std::make_shared<ApduResponse>(
547  rapdu, apduRequest->getSuccessfulStatusCodes());
548 
549  if (apduRequest->isCase4() && apduResponse->getDataOut().empty() &&
550  apduResponse->isSuccessful()) {
551  /* Do the get response command but keep the original status code */
552  apduResponse = case4HackGetResponse(apduResponse->getStatusCode());
553  }
554 
555  timeStamp = System::nanoTime();
556  elapsed10ms = (timeStamp - mBefore) / 100000;
557  mBefore = timeStamp;
558 
559  mLogger->debug("[%] processApduRequest => %, elapsed % ms\n",
560  getName(), apduResponse, elapsed10ms / 10);
561 
562  return apduResponse;
563 }

565 std::shared_ptr<ApduResponse> AbstractLocalReader::case4HackGetResponse(int originalStatusCode)
566 {
567  mLogger->debug("case4HackGetResponse\n");
568 
569  /*
570  * Build a get response command the actual length expected by the SE in the
571  * get response command is handled in transmitApdu
572  */
573  long long timeStamp = System::nanoTime();
574  long long elapsed10ms = (timeStamp - mBefore) / 100000;
575  mBefore = timeStamp;
576 
577  mLogger->debug("[%] case4HackGetResponse => ApduRequest: NAME = \"Internal Get Response\", "
578  "RAWDATA = %, elapsed = %\n",
579  getName(),
580  mGetResponseHackRequestBytes,
581  elapsed10ms / 10.0);
582 
583  std::vector<uint8_t> getResponseHackResponseBytes =
584  transmitApdu(mGetResponseHackRequestBytes);
585 
586  /*
587  * We expect here a 0x9000 status code
588  *
589  * C++ vs. Java: Careful, Java modifies the byte array *after* having created the ApduResponse
590  * class - does not work at all in C++. Create a temp ApduRespon object, do the
591  * buffer modification if necessary and create the final object.
592  */
593  std::shared_ptr<ApduResponse> getResponseHackResponseTemp =
594  std::make_shared<ApduResponse>(getResponseHackResponseBytes, nullptr);
595 
596  timeStamp = System::nanoTime();
597  elapsed10ms = (timeStamp - mBefore) / 100000;
598  mBefore = timeStamp;
599  mLogger->debug("[%] case4HackGetResponse => Internal %, elapsed % ms\n",
600  getName(),
601  mGetResponseHackRequestBytes,
602  elapsed10ms / 10.0);
603 
604  if (getResponseHackResponseTemp->isSuccessful()) {
605  /* Replace the two last status word bytes by the original status word */
606  getResponseHackResponseBytes[getResponseHackResponseBytes.size() - 2] =
607  static_cast<uint8_t>((originalStatusCode & 0xff00) >> 8);
608  getResponseHackResponseBytes[getResponseHackResponseBytes.size() - 1] =
609  static_cast<uint8_t>(originalStatusCode & 0x00ff);
610  }
611 
612  return std::make_shared<ApduResponse>(getResponseHackResponseBytes, nullptr);
613 }

628 std::shared_ptr<ApduResponse> AbstractLocalReader::processExplicitAidSelection(
629  SeSelector::AidSelector& aidSelector)
630 {
631  std::shared_ptr<ApduResponse> fciResponse;
632  const std::vector<uint8_t> aid = aidSelector.getAidToSelect();
633 
634  if (aid.empty())
635  throw IllegalArgumentException("AID must not be null for an AidSelector");
636 
637  mLogger->debug("[%] openLogicalChannel => Select Application with AID = %\n", getName(), aid);
638 
639  /*
640  * build a get response command the actual length expected by the SE in the
641  * get response command is handled in transmitApdu
642  */
643  std::vector<uint8_t> selectApplicationCommand;
644  selectApplicationCommand.push_back(0x00); // CLA
645  selectApplicationCommand.push_back(0xA4); // INS
646  selectApplicationCommand.push_back(0x04); // P1: select by name
647 
648  /*
649  * P2: b0,b1 define the File occurrence, b2,b3 define the File control
650  * information. We use the bitmask defined in the respective enums.
651  */
652  selectApplicationCommand.push_back(aidSelector.getFileOccurrence().getIsoBitMask() |
653  aidSelector.getFileControlInformation().getIsoBitMask());
654  selectApplicationCommand.push_back((uint8_t)(aid.size())); // Lc
655  selectApplicationCommand.insert(selectApplicationCommand.end(), aid.begin(), aid.end()); // Data
656  selectApplicationCommand.push_back(0x00); // Le
657 
658  /*
659  * We use here processApduRequest to manage case 4 hack. The successful status codes list for
660  * this command is provided.
661  */
662  fciResponse = processApduRequest(std::make_shared<ApduRequest>(
663  "Internal Select Application",
664  selectApplicationCommand,
665  true,
666  aidSelector.getSuccessfulSelectionStatusCodes()));
667 
668  if (!fciResponse->isSuccessful()) {
669  mLogger->debug("[%] openLogicalChannel => Application Selection failed. SELECTOR = %\n",
670  getName(),
671  aidSelector);
672  }
673 
674  return fciResponse;
675 }

677 std::shared_ptr<ApduResponse> AbstractLocalReader::recoverSelectionFciData(
678  SeSelector::AidSelector& aidSelector)
679 {
680  std::shared_ptr<ApduResponse> fciResponse;
681 
682  /*
683  * Get Data APDU: CLA, INS, P1: always 0, P2: 0x6F FCI for the current DF,
684  * LC: 0
685  */
686  std::vector<uint8_t> getDataCommand = {0x00, 0xCA, 0x00, 0x6F, 0x00};
687 
688  /* The successful status codes list for this command is provided */
689  fciResponse = processApduRequest(std::make_shared<ApduRequest>(
690  "Internal Get Data",
691  getDataCommand,
692  false,
693  aidSelector.getSuccessfulSelectionStatusCodes()));
694 
695  if (!fciResponse->isSuccessful()) {
696  mLogger->debug("[%] selectionGetData => Get data failed. SELECTOR = "
697  "%\n", getName(), aidSelector);
698  }
699 
700  return fciResponse;
