241 static std::shared_ptr<SeSelector> getSelector(
242  std::shared_ptr<std::set<int>> selectionStatusCode)
243 {
244  /*
245  * We can use a fake AID here because it is not fully interpreted, the
246  * purpose of this unit test is to verify the proper format of the request.
247  */
248  const std::vector<uint8_t> aid = {0xAA, 0xBB, 0xCC, 0xDD, 0xEE, 0xFF};
249 
250  std::shared_ptr<SeSelector::AidSelector> aidSelector =
251  SeSelector::AidSelector::builder()
252  ->aidToSelect(aid)
253  .build();
254 
255  if (selectionStatusCode) {
256  for (const auto& code : *selectionStatusCode.get())
257  aidSelector->addSuccessfulStatusCode(code);
258  }
259 
260  return SeSelector::builder()
261  ->seProtocol(SeCommonProtocols::PROTOCOL_ISO14443_4)
262  .atrFilter(nullptr)
263  .aidSelector(aidSelector)
264  .build();
265 }
