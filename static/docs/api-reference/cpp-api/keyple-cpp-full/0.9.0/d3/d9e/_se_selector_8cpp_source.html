196 {
197  bool m;
198 
199  if (mAtrRegex.length() != 0) {
200  std::unique_ptr<Pattern> p = Pattern::compile(mAtrRegex);
201  std::string atrString = ByteArrayUtil::toHex(atr);

203  } else {
