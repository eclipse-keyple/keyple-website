189 static std::shared_ptr<SeSelector> getSelector(
190  std::shared_ptr<std::set<int>> selectionStatusCode)
191 {
192  /*
193  * We can use a fake AID here because it is not fully interpreted, the
194  * purpose of this unit test is to verify the proper format of the request.

197 
198  std::shared_ptr<SeSelector::AidSelector> aidSelector =
199  SeSelector::AidSelector::builder()
200  ->aidToSelect(aid)
201  .build();
202 
203  for (const auto& code : *selectionStatusCode.get())
204  aidSelector->addSuccessfulStatusCode(code);
205 
206  return SeSelector::builder()
207  ->aidSelector(aidSelector)
208  .atrFilter(nullptr)
209  .seProtocol(SeCommonProtocols::PROTOCOL_ISO14443_4)
210  .build();
