113 void* AbstractThreadedObservablePlugin::EventThread::run()
114 {
115  mParent->mLogger->trace("starting event thread\n");
116 
117  auto changedReaderNames = std::make_shared<std::set<std::string>>();
118 
119  try {
120  while ((!isInterrupted()) && mRunning) {
121  /* Retrieves the current readers names list */
122  mParent->mLogger->debug("run - fetching native readers names\n");
123  const std::set<std::string>& actualNativeReadersNames = mParent->fetchNativeReadersNames();
124 
125  /* Checks if it has changed this algorithm favors cases where nothing changes */
126  if (mParent->mNativeReadersNames != actualNativeReadersNames) {
127  mParent->mLogger->debug("run - mNativeReadersNames != actualNativeReadersNames\n");
128 
129  /*
130  * Parse the current readers list, notify for disappeared readers, update readers
131  * list
132  */
133 
134  /* Build changed reader names list */
135  changedReaderNames->clear();
136 
137  mParent->mLogger->debug("run - building changedReaderNames list\n");
138  for (const auto& pair : mParent->mNativeReaders) {
139  const auto& it = actualNativeReadersNames.find(pair.second->getName());
140  if (it == actualNativeReadersNames.end())
141  changedReaderNames->insert(pair.second->getName());
142  }
143 
144  /* Notify disconnections if any and update the reader list */
145  if (changedReaderNames->size() > 0) {
146  mParent->mLogger->debug("run - changedReaderNames not empty\n");
147 
148  /* Grouped notification */
149  auto event = std::make_shared<PluginEvent>(mPluginName,
150  changedReaderNames,
151  EventType::READER_DISCONNECTED);
152  mParent->notifyObservers(event);
153 
154  /* list update */
155  for (auto it = mParent->mNativeReaders.begin();
156  it != mParent->mNativeReaders.end();) {
157  const std::shared_ptr<SeReader> reader = it->second;
158 
159  if (actualNativeReadersNames.find(reader->getName()) ==
160  actualNativeReadersNames.end()) {
161  /* Removes any possible observers before removing the reader */
162  auto observable = std::dynamic_pointer_cast<ObservableReader>(reader);
163  if (observable) {
164  observable->clearObservers();
165  /* In case where Reader was detecting SE */
166  observable->stopSeDetection();
167  }
168 
169  mParent->mLogger->trace("[%][%] Plugin thread => Remove unplugged " \
170  "reader from readers list\n",
171  mPluginName,
172  reader->getName());
173 
174  /* Remove reader name from the current list */
175  mParent->mNativeReadersNames.erase(reader->getName());
176 
177  /* C++ vs. Java: remove reader *after* we are done with it */
178  mParent->mNativeReaders.erase(it++);
179  } else {
180  ++it;
181  }
182  }
183 
184  /* Clean the list for a possible connection notification */
185  changedReaderNames->clear();
186  }
187 
188  /* Parse the new readers list, notify for readers appearance, update readers list */
189  for (const auto& readerName : actualNativeReadersNames) {
190  const auto it = mParent->mNativeReadersNames.find(readerName);
191  if (it == mParent->mNativeReadersNames.end()) {
192  std::shared_ptr<SeReader> reader = mParent->fetchNativeReader(readerName);
193  mParent->mNativeReaders.insert({reader->getName(), reader});
194 
195  /* Add to the notification list */
196  changedReaderNames->insert(readerName);
197  mParent->mLogger->trace("[%][%] Plugin thread => Add plugged reader to " \
198  "readers list\n",
199  mPluginName,
200  reader->getName());
201 
202  /* add reader name to the current list */
203  mParent->mNativeReadersNames.insert(readerName);
204  }
205  }
206 
207  /* notify connections if any */
208  if (changedReaderNames->size() > 0) {
209  auto event = std::make_shared<PluginEvent>(mPluginName,
210  changedReaderNames,
211  EventType::READER_CONNECTED);
212  mParent->notifyObservers(event);
213  }
214  }
215 
216  /* sleep for a while. */
217  Thread::sleep((long)mParent->mThreadWaitTimeout);
218  }
219 
220  } catch (const InterruptedException& e) {
221  mParent->mLogger->warn("[%] An exception occurred while monitoring plugin: %\n",
222  mPluginName,
223  e);
224 
225  /* Restore interrupted state */
226  //Thread::currentThread().interrupt();
227  } catch (const KeypleReaderException& e) {
228  mParent->mLogger->warn("[%] An exception occurred while monitoring plugin: %\n",
229  mPluginName,
230  e);
231  }
232 
233  return nullptr;
234 }
