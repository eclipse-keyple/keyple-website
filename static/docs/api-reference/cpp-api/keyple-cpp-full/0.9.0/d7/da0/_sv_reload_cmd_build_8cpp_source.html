76 void SvReloadCmdBuild::finalizeBuilder(const std::vector<uint8_t>& reloadComplementaryData)
77 {
78  if ((mPoRevision == PoRevision::REV3_2 && reloadComplementaryData.size() != 20) ||
79  (mPoRevision != PoRevision::REV3_2 && reloadComplementaryData.size() != 15))
80  throw IllegalArgumentException("Bad SV prepare load data length.");
81 
82  const uint8_t p1 = reloadComplementaryData[4];
83  const uint8_t p2 = reloadComplementaryData[5];
84 
85  mDataIn[0] = reloadComplementaryData[6];
86  System::arraycopy(reloadComplementaryData, 0, mDataIn, 11, 4);
