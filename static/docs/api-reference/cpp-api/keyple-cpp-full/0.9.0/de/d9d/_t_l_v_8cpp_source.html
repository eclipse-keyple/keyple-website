39 bool TLV::parse(const std::shared_ptr<Tag> tag, int offset)
40 {
41  if (tag == nullptr) {
42  throw std::invalid_argument("TLV parsing: tag can't be null.");
43  }
44 
45  try {
46  mTag = std::make_shared<Tag>(mBinary, offset);
47  } catch (const std::out_of_range& e) {
48  (void)e;
49  throw std::invalid_argument("TLV parsing: index is too large.");
50  }
51 
52  mLength = 0;
53  if (*tag.get() == *(mTag.get())) {
54  offset += mTag->getTagSize();
55  mPosition += mTag->getTagSize();
56  if ((mBinary[offset] & 0x80) == 0x00) {
57  /* short form: single octet length */
58  mLength += static_cast<int>(mBinary[offset]);
59  mPosition++;
60  } else {
61  /*
62  * Long form: first octet (b6-b0)) gives the number of following
63  * length octets
64  */
65  int following = (mBinary[offset] & 0x7F);
66  mPosition++;
67  while (following > 0) {
68  offset++;
69  mPosition++;
70  mLength += (mBinary[offset] & 0xFF);
71  if (following > 1) {
72  mLength <<= 8;
73  }
74  following--;
75  }
76  }
77  return true;
78  } else {
79  return false;
80  }
81 }
