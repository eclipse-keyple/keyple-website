54  switch (selectFileControl) {

64  p1 = 0x09;

66  break;
67  default:
68  throw IllegalStateException("Unsupported selectFileControl parameter " +
69  StringHelper::toString(selectFileControl));
70  }
71 
72  mRequest = setApduRequest(cla, command, p1, p2, selectData, 0x00);
73 
74  const std::string extraInfo = "SELECTIONCONTROL=" + StringHelper::toString(selectFileControl);
