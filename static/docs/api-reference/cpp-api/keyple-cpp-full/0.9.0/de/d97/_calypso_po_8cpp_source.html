73 CalypsoPo::CalypsoPo(std::shared_ptr<SeResponse> selectionResponse,
74  const TransmissionMode& transmissionMode)
75 : AbstractMatchingSe(selectionResponse, transmissionMode),
76  mIsDfRatified(false)
77 {
78  if (hasFci() && static_cast<int>(getFciBytes().size()) > 2) {
79 
80  /*
81  * Parse PO FCI - to retrieve DF Name (AID), Serial Number, &amp;
82  * StartupInfo
83  */
84  GetDataFciRespPars poFciRespPars(
85  selectionResponse->getSelectionStatus()->getFci(), nullptr);
86 
87  /* 4 fields extracted by the low level parser */
88  mDfName = poFciRespPars.getDfName();
89  mCalypsoSerialNumber = poFciRespPars.getApplicationSerialNumber();
90  mStartupInfo = poFciRespPars.getDiscretionaryData();
91  mIsDfInvalidated = poFciRespPars.isDfInvalidated();
92 
93  const uint8_t applicationType = getApplicationType();
94  mRevision = determineRevision(applicationType);
95 
96  /* Session buffer size */
97  int bufferSizeIndicator = mStartupInfo[SI_BUFFER_SIZE_INDICATOR];
98  int bufferSizeValue = BUFFER_SIZE_INDICATOR_TO_BUFFER_SIZE[bufferSizeIndicator];
99 
100  if (mRevision == PoRevision::REV2_4) {
101  /*
102  * Old cards have their modification counter in number of commands
103  */
104  mModificationCounterIsInBytes = false;
105  mModificationsCounterMax =
106  REV2_PO_DEFAULT_WRITE_OPERATIONS_NUMBER_SUPPORTED_PER_SESSION;
107  } else {
108  mModificationsCounterMax = bufferSizeValue;
109  }
110 
111  mIsConfidentialSessionModeSupported =
112  (applicationType & APP_TYPE_CALYPSO_REV_32_MODE) != 0;
113  mIsDeselectRatificationSupported =
114  (applicationType & APP_TYPE_RATIFICATION_COMMAND_REQUIRED) == 0;
115  mIsSvFeatureAvailable =
116  (applicationType & APP_TYPE_WITH_CALYPSO_SV) != 0;
117  mIsPinFeatureAvailable =
118  (applicationType & APP_TYPE_WITH_CALYPSO_PIN) != 0;
119  mIsPublicAuthenticationSupported =
120  (applicationType & APP_TYPE_WITH_PUBLIC_AUTHENTICATION) != 0;
121  } else {
122  /*
123  * FCI is not provided: we consider it is Calypso PO rev 1, it's serial
124  * number is provided in the ATR
125  */
126  if (!hasAtr())
127  throw IllegalStateException("Unable to identify this PO: Neither the CFI nor the " \
128  "ATR are available.");
129 
130  const std::vector<uint8_t> atr = getAtrBytes();
131 
132  /*
133  * Basic check: we expect to be here following a selection based on the
134  * ATR
135  */
136  if (static_cast<int>(atr.size()) != PO_REV1_ATR_LENGTH)
137  throw IllegalStateException("Unexpected ATR length: " +
138  ByteArrayUtil::toHex(getAtrBytes()));
139 
140  mRevision = PoRevision::REV1_0;
141 
142  /* C++: this line below seems useless */
143  //mDfName = null;
144  mCalypsoSerialNumber = std::vector<uint8_t>(8);
145 
146 
147  /* Old cards have their modification counter in number of commands */
148  mModificationCounterIsInBytes = false;
149 
150  /* The array is initialized with 0 (cf. default value for primitive types) */
151  System::arraycopy(atr, 12, mCalypsoSerialNumber, 4, 4);
152  mModificationsCounterMax = REV1_PO_DEFAULT_WRITE_OPERATIONS_NUMBER_SUPPORTED_PER_SESSION;
153 
154  mStartupInfo = std::vector<uint8_t>(7);
155  /* Create buffer size indicator */
156  mStartupInfo[0] = static_cast<uint8_t>(mModificationsCounterMax);
157  /* Create the startup info with the 6 bytes of the ATR from position 6 */
158  System::arraycopy(atr, 6, mStartupInfo, 1, 6);
159 
160  /* TODO check these flags */
161  mIsConfidentialSessionModeSupported = false;
162  mIsDeselectRatificationSupported = true;
163  mIsSvFeatureAvailable = false;
164  mIsPinFeatureAvailable = false;
165  mIsPublicAuthenticationSupported = false;
166  mIsDfInvalidated = false;
167  }
168 
169  /*
170  * Rev1 and Rev2 expects the legacy class byte while Rev3 expects the ISO
171  * class byte
172  */
173  if (mRevision == PoRevision::REV1_0 || mRevision == PoRevision::REV2_4)
174  mPoClass = std::make_shared<PoClass>(PoClass::LEGACY);
175  else
176  mPoClass = std::make_shared<PoClass>(PoClass::ISO);
177 }

464 PoRevision CalypsoPo::determineRevision(const uint8_t applicationType) const
465 {
466  if ((applicationType & (1 << 7)) != 0) {
467  /* CLAP */
468  return PoRevision::REV3_1_CLAP;
469  } else if ((applicationType >> 3) == 0x05) {
470  return PoRevision::REV3_2;
471  } else if ((applicationType >> 3) == 0x04) {
472  return PoRevision::REV3_1;
473  } else {
474  return PoRevision::REV2_4;
475  }
476 }
