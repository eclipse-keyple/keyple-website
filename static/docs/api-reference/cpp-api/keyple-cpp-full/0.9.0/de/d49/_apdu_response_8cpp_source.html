31 ApduResponse::ApduResponse(
32  const std::vector<uint8_t>& buffer,
33  const std::shared_ptr<std::set<int>> successfulStatusCodes)
34 : mBytes(buffer)
35 {
36  if (buffer.empty()) {
37  mSuccessful = false;
38  } else {
39  if (buffer.size() < 2)
40  throw std::invalid_argument("Bad buffer (length < 2): " +
41  std::to_string(buffer.size()));
42 
43  int statusCode = ((buffer[buffer.size() - 2] & 0x000000FF) << 8) +
44  (buffer[buffer.size() - 1] & 0x000000FF);
45 
46  if (successfulStatusCodes != nullptr)
47  mSuccessful = statusCode == 0x9000 ||
48  (successfulStatusCodes->find(statusCode) !=
49  successfulStatusCodes->end());
50  else
51  mSuccessful = statusCode == 0x9000;
52  }
53 }
