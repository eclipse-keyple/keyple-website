57 std::shared_ptr<SeResource<CalypsoSam>> SamResourceManagerPool::allocateSamResource(
58  const AllocationMode allocationMode,
59  const std::shared_ptr<SamIdentifier> samIdentifier)
60 {
61  const uint64_t maxBlockingDate = System::currentTimeMillis() + mMaxBlockingTime;
62  bool noSamResourceLogged = false;
63 
64  mLogger->debug("Allocating SAM reader channel...\n");
65 
66  while (true) {
67  try {
68  /* virtually infinite number of readers */
69  std::shared_ptr<SeReader> samReader = mSamReaderPlugin.allocateReader(
70  samIdentifier->getGroupReference());
71 
72  if (samReader != nullptr) {
73  std::shared_ptr<ManagedSamResource> managedSamResource = createSamResource(samReader);
74  mLogger->debug("Allocation succeeded. SAM resource created\n");
75  return managedSamResource;
76  }
77 
78  /*
79  * Loop until MAX_BLOCKING_TIME in blocking mode, only once in
80  * non-blocking mode
81  */
82  if (allocationMode == AllocationMode::NON_BLOCKING) {
83  mLogger->trace("No SAM resources available at the moment\n");
84  throw CalypsoNoSamResourceAvailableException(
85  "No Sam resource could be allocated for " \
86  "samIdentifier " +
87  samIdentifier->getGroupReference());
88  } else {
89  if (!noSamResourceLogged) {
90  /* log once the first time */
91  mLogger->trace("No SAM resources available at the moment\n");
92  noSamResourceLogged = true;
93  }
94 
95  try {
96  Thread::sleep(mSleepTime);
97  } catch (const InterruptedException& e) {
98  /*
99  * C++ vs. Java: Cannot interrupt thread easily. This
100  * scenario is unlikely to happen so throw
101  * for now.
102  */
103  (void)e;
104  //Thread::currentThread()->interrupt();
105  mLogger->error("Interrupt exception in Thread::sleep\n");
106  throw;
107  }
108 
109  if (System::currentTimeMillis() >=
110  static_cast<unsigned long long>(maxBlockingDate)) {
111  mLogger->error("The allocation process failed. Timeout % " \
112  "sec exceeded\n",
113  (mMaxBlockingTime / 1000.0));
114  std::stringstream ss;
115  ss << "No Sam resource could be allocated"
116  << "within " << mMaxBlockingTime
117  << "ms for samIdentifier "
118  << samIdentifier->getGroupReference();
119  throw CalypsoNoSamResourceAvailableException(ss.str());
120  }
121  }
122 
123  } catch (const KeypleAllocationReaderException& e) {
124  throw KeypleAllocationReaderException(
125  "Allocation failed due to a plugin technical error", e);
126  } catch (const KeypleAllocationNoReaderException& e) {
127  /* No reader is available, let's retry */
128  (void)e;
129  }
130  }
131 }
