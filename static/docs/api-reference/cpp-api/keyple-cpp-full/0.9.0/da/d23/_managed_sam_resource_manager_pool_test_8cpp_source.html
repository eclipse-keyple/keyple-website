192 TEST(ManagedSamResourceManagerPoolTest, waitResources)
193 {
194  /* Init */
195  MSRMP_ReaderPoolPluginMock poolPlugin;
196  SamResourceManagerPool srmSpy(poolPlugin, 1000, 10);
197  long long start = System::currentTimeMillis();
198  bool exceptionThrown = false;
199 
200  EXPECT_CALL(poolPlugin, allocateReader(_)).WillRepeatedly(Return(nullptr));
201 
202  /* Test */
203  std::shared_ptr<SeResource<CalypsoSam>> out = nullptr;
204  try {
205  auto samIdentifier = SamIdentifier::builder()->samRevision(SamRevision::AUTO)
206  .serialNumber("any")
207  .groupReference("any")
208  .build();
209  out = srmSpy.allocateSamResource(AllocationMode::BLOCKING, samIdentifier);
210  } catch (const CalypsoNoSamResourceAvailableException& e) {
211  (void)e;
212  exceptionThrown = true;
213  } catch (const KeypleAllocationReaderException& e) {
214  (void)e;
215  exceptionThrown = true;
216  }
217 
218  long long stop = System::currentTimeMillis();
219 
220  /* Assert an exception is thrown after MAX_BLOCKING_TIME */
221  ASSERT_TRUE(exceptionThrown);
222  ASSERT_GT(stop - start, 1000);
223 }
