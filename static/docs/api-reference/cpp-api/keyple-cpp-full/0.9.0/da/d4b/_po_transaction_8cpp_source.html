146 void PoTransaction::processAtomicOpening(
147  const AccessLevel& accessLevel,
148  std::vector<std::shared_ptr<AbstractPoCommandBuilder<AbstractPoResponseParser>>>& poCommands)
149 {
150  /* This method should be called only if no session was previously open */
151  checkSessionIsNotOpen();
152 
153  if (mPoSecuritySettings == nullptr)
154  throw CalypsoPoTransactionIllegalStateException("No SAM resource is available");
155 
156  /* Gets the terminal challenge */
157  const std::vector<uint8_t>& sessionTerminalChallenge =
158  mSamCommandProcessor->getSessionTerminalChallenge();
159 
160  /* PO ApduRequest List to hold Open Secure Session and other optional commands */
161  std::vector<std::shared_ptr<ApduRequest>> poApduRequests;
162 
163  /*
164  * The sfi and record number to be read when the open secure session command is executed. The
165  * default value is 0 (no record to read) but we will optimize the exchanges if a read record
166  * command has been prepared.
167  */
168  uint8_t sfi = 0;
169  uint8_t recordNumber = 0;
170 
171  /*
172  * Let's check if we have a read record command at the top of the command list.
173  *
174  * If so, then the command is withdrawn in favour of its equivalent executed at the same time as
175  * the open secure session command.
176  */
177  if (!poCommands.empty()) {
178  std::shared_ptr<AbstractPoCommandBuilder<AbstractPoResponseParser>> poCommand = poCommands[0];
179  auto readRecCmdBuild = std::reinterpret_pointer_cast<ReadRecordsCmdBuild>(poCommand);
180  auto commandRef = std::dynamic_pointer_cast<CalypsoPoCommand>(poCommand->getCommandRef());
181 
182  if (*commandRef.get() == CalypsoPoCommand::READ_RECORDS &&
183  readRecCmdBuild->getReadMode() == ReadRecordsCmdBuild::ReadMode::ONE_RECORD) {
184  sfi = readRecCmdBuild->getSfi();
185  recordNumber = readRecCmdBuild->getFirstRecordNumber();
186  poCommands.erase(poCommands.begin());
187  }
188  }
189 
190  /* Build the PO Open Secure Session command */
191  auto openSessionCmdBuild = AbstractOpenSessionCmdBuild<AbstractOpenSessionRespPars>::create(
192  mCalypsoPo->getRevision(),
193  accessLevel.getSessionKey(),
194  sessionTerminalChallenge,
195  sfi,
196  recordNumber);
197 
198  /* Add the resulting ApduRequest to the PO ApduRequest list */
199  poApduRequests.push_back(openSessionCmdBuild->getApduRequest());
200 
201  /* Add all optional commands to the PO ApduRequest list */
202  const std::vector<std::shared_ptr<ApduRequest>>& commands = getApduRequests(poCommands);
203  std::for_each(commands.begin(),
204  commands.end(),
205  [&](const std::shared_ptr<ApduRequest>& r) {poApduRequests.push_back(r);});
206 
207  /* Create a SeRequest from the ApduRequest list, PO AID as Selector, keep channel open */
208  auto poSeRequest = std::make_shared<SeRequest>(poApduRequests);
209 
210  /* Transmit the commands to the PO */
211  std::shared_ptr<SeResponse> poSeResponse = safePoTransmit(poSeRequest,
212  ChannelControl::KEEP_OPEN);
213 
214 
215  /* Retrieve and check the ApduResponses */
216  std::vector<std::shared_ptr<ApduResponse>>& poApduResponses = poSeResponse->getApduResponses();
217 
218  /* Do some basic checks */
219  checkCommandsResponsesSynchronization(poApduRequests.size(), poApduResponses.size());
220 
221  /*
222  * Parse the response to Open Secure Session (the first item of poApduResponses). The
223  * updateCalypsoPo method fills the CalypsoPo object with the command data and return the parser
224  * used for an internal usage here.
225  */
226  auto poOpenSessionPars =
227  std::dynamic_pointer_cast<AbstractOpenSessionRespPars>(
228  CalypsoPoUtils::updateCalypsoPo(
229  mCalypsoPo,
230  std::reinterpret_pointer_cast<AbstractPoCommandBuilder<AbstractPoResponseParser>>(
231  openSessionCmdBuild),
232  poApduResponses[0]));
233 
234  /*
235  * Build the Digest Init command from PO Open Session the session challenge is needed for the
236  * SAM digest computation
237  */
238  const std::vector<uint8_t>& sessionCardChallenge = poOpenSessionPars->getPoChallenge();
239 
240  /* The PO KIF */
241  const uint8_t poKif = poOpenSessionPars->getSelectedKif();
242 
243  /* The PO KVC, may be null for PO Rev 1.0 */
244  const uint8_t poKvc = poOpenSessionPars->getSelectedKvc();
245 
246  logger->debug("processAtomicOpening => opening: CARDCHALLENGE = %, POKIF = %, POKVC = %\n",
247  ByteArrayUtil::toHex(sessionCardChallenge),
248  StringHelper::uint8ToHexString(poKif),
249  StringHelper::uint8ToHexString(poKvc));
250 
251  if (!poKvc || !mPoSecuritySettings->isSessionKvcAuthorized(poKvc))
252  throw CalypsoUnauthorizedKvcException("PO KVC = " + StringHelper::uint8ToHexString(poKvc));
253 
254  /*
255  * Initialize the digest processor. It will store all digest operations
256  * (Digest Init, Digest Update) until the session closing. At this moment,
257  * all SAM Apdu will be processed at once.
258  */
259  mSamCommandProcessor->initializeDigester(accessLevel,
260  false,
261  false,
262  poKif,
263  poKvc,
264  poApduResponses[0]->getDataOut());
265 
266  /*
267  * Add all commands data to the digest computation. The first command in the
268  * list is the open secure session command. This command is not included in
269  * the digest computation, so we skip it and start the loop at index 1.
270  */
271  if (!poCommands.empty())
272  /* Add requests and responses to the digest processor */
273  mSamCommandProcessor->pushPoExchangeDataList(poApduRequests,
274  poApduResponses,
275  1);
276 
277  /* Remove Open Secure Session response and create a new SeResponse */
278  poApduResponses.erase(poApduResponses.begin());
279 
280  /* Update CalypsoPo with the received data */
281  CalypsoPoUtils::updateCalypsoPo(mCalypsoPo, poCommands, poApduResponses);
282 
283  mSessionState = SessionState::SESSION_OPEN;
284 }

343 void PoTransaction::processAtomicClosing(
344  const std::vector<std::shared_ptr<AbstractPoCommandBuilder< AbstractPoResponseParser>>>&
345  poModificationCommands,
346  const std::vector<std::shared_ptr<ApduResponse>>& poAnticipatedResponses,
347  const SessionSetting::RatificationMode ratificationMode,
348  const ChannelControl channelControl)
349 {
350  checkSessionIsOpen();
351 
352  /* Get the PO ApduRequest List - for the first PO exchange */
353  std::vector<std::shared_ptr<ApduRequest>> poApduRequests =
354  getApduRequests(poModificationCommands);
355 
356  /*
357  * Compute "anticipated" Digest Update (for optional poModificationCommands)
358  */
359  if (!poModificationCommands.empty() && !poApduRequests.empty()) {
360  checkCommandsResponsesSynchronization(poApduRequests.size(), poAnticipatedResponses.size());
361 
362  /* Add all commands data to the digest computation: commands and anticipated responses */
363  mSamCommandProcessor->pushPoExchangeDataList(poApduRequests, poAnticipatedResponses, 0);
364  }
365 
366  /*
367  * All SAM digest operations will now run at once.
368  * Get Terminal Signature from the latest response
369  */
370  const std::vector<uint8_t>& sessionTerminalSignature =
371  mSamCommandProcessor->getTerminalSignature();
372 
373  bool ratificationCommandResponseReceived;
374 
375  /* Build the PO Close Session command. The last one for this session */
376  auto closeSessionCmdBuild = std::make_shared<CloseSessionCmdBuild>(
377  mCalypsoPo->getPoClass(),
378  RatificationMode::CLOSE_RATIFIED == ratificationMode,
379  sessionTerminalSignature);
380 
381  poApduRequests.push_back(closeSessionCmdBuild->getApduRequest());
382 
383  /* Keep the position of the Close Session command in request list */
384  const int closeCommandIndex = poApduRequests.size() - 1;
385 
386  /* Add the PO Ratification command if any */
387  bool ratificationCommandAdded;
388 
389  if (ratificationMode == RatificationMode::CLOSE_RATIFIED &&
390  mCalypsoPo->getTransmissionMode() == TransmissionMode::CONTACTLESS) {
391  poApduRequests.push_back(RatificationCmdBuild::getApduRequest(mCalypsoPo->getPoClass()));
392  ratificationCommandAdded = true;
393  } else {
394  ratificationCommandAdded = false;
395  }
396 
397  /* Transfer PO commands */
398  auto poSeRequest = std::make_shared<SeRequest>(poApduRequests);
399  std::shared_ptr<SeResponse> poSeResponse;
400 
401  try {
402  poSeResponse = mPoReader->transmitSeRequest(poSeRequest, channelControl);
403 
404  /*
405  * If the ratification command was added and no error occured then the response has been
406  * received
407  */
408  ratificationCommandResponseReceived = ratificationCommandAdded;
409  } catch (const KeypleReaderIOException& e) {
410  poSeResponse = e.getSeResponse();
411 
412  /*
413  * The current exception may have been caused by a communication issue with the PO during
414  * the ratification command.
415  *
416  * In this case, we do not stop the process and consider the Secure Session close. We'll
417  * check the signature.
418  *
419  * We should have one response less than requests.
420  */
421  if (!ratificationCommandAdded ||
422  poSeResponse == nullptr ||
423  poSeResponse->getApduResponses().size() != poApduRequests.size() - 1)
424  throw CalypsoPoIOException("PO IO Exception while transmitting commands.", e);
425 
426  /* We received all responses except the response to the ratification command */
427  ratificationCommandResponseReceived = false;
428  }
429 
430  std::vector<std::shared_ptr<ApduResponse>> poApduResponses = poSeResponse->getApduResponses();
431 
432  /*
433  * Check the commands executed before closing the secure session (only responses to these
434  * commands will be taken into account)
435  */
436  CalypsoPoUtils::updateCalypsoPo(mCalypsoPo, poModificationCommands, poApduResponses);
437 
438  /* Check the PO's response to Close Secure Session */
439  std::shared_ptr<CloseSessionRespPars> poCloseSessionPars;
440 
441  try {
442  auto cmd = std::reinterpret_pointer_cast<AbstractPoCommandResponse>(closeSessionCmdBuild);
443  std::shared_ptr<AbstractPoResponseParser> parser =
444  CalypsoPoUtils::updateCalypsoPo(mCalypsoPo, cmd, poApduResponses[closeCommandIndex]);
445  poCloseSessionPars = std::dynamic_pointer_cast<CloseSessionRespPars>(parser);
446  } catch (const CalypsoPoSecurityDataException& ex) {
447  throw CalypsoPoCloseSecureSessionException("Invalid PO session", ex);
448  }
449 
450  /* Check the PO signatures */
451  try {
452  mSamCommandProcessor->authenticatePoSignature(poCloseSessionPars->getSignatureLo());
453  } catch (const CalypsoSamIOException& e) {
454  throw CalypsoAuthenticationNotVerifiedException(e.getMessage());
455  } catch (const CalypsoSamCommandException& e) {
456  throw CalypsoSessionAuthenticationException("PO authentication failed on SAM side.", e);
457  }
458 
459  /* If necessary, we check the status of the SV after the session has been successfully closed */
460  if (mPoCommandManager->isSvOperationCompleteOneTime())
461  mSamCommandProcessor->checkSvStatus(poCloseSessionPars->getPostponedData());
462 
463  mSessionState = SessionState::SESSION_CLOSED;
464 
465  if (ratificationCommandResponseReceived)
466  /* Remove the ratification response */
467  poApduResponses.pop_back();
468 
469  /* Remove Close Secure Session response and create a new SeResponse */
470  poApduResponses.pop_back();
471 }

526 std::vector<std::shared_ptr<ApduResponse>> PoTransaction::getAnticipatedResponses(
527  const std::vector<std::shared_ptr<AbstractPoCommandBuilder<AbstractPoResponseParser>>>&
528  poCommands) const
529 {
530  std::vector<std::shared_ptr<ApduResponse>> apduResponses;
531 
532  for (const auto& commandBuilder : poCommands) {
533  std::shared_ptr<SeCommand> cmd = commandBuilder->getCommandRef();
534  auto calypsoCmd = std::dynamic_pointer_cast<CalypsoPoCommand>(cmd);
535  if (*calypsoCmd.get() == CalypsoPoCommand::DECREASE) {
536  auto decBuilder = std::reinterpret_pointer_cast<DecreaseCmdBuild>(commandBuilder);
537  const uint8_t sfi = decBuilder->getSfi();
538  const int counter = decBuilder->getCounterNumber();
539  const int newCounterValue = getCounterValue(sfi, counter) - decBuilder->getDecValue();
540  apduResponses.push_back(createIncreaseDecreaseResponse(newCounterValue));
541 
542  } else if (*calypsoCmd.get() == CalypsoPoCommand::INCREASE) {
543  auto incBuilder = std::reinterpret_pointer_cast<IncreaseCmdBuild>(commandBuilder);
544  const uint8_t sfi = incBuilder->getSfi();
545  const int counter = incBuilder->getCounterNumber();
546  const int newCounterValue = getCounterValue(sfi, counter) + incBuilder->getIncValue();
547  apduResponses.push_back(createIncreaseDecreaseResponse(newCounterValue));
548 
549  } else if (*calypsoCmd.get() == CalypsoPoCommand::SV_RELOAD ||
550  *calypsoCmd.get() == CalypsoPoCommand::SV_DEBIT ||
551  *calypsoCmd.get() == CalypsoPoCommand::SV_UNDEBIT) {
552  apduResponses.push_back(RESPONSE_OK_POSTPONED);
553  } else {
554  /* Append/Update/Write Record: response = 9000 */
555  apduResponses.push_back(RESPONSE_OK);
556  }
557  }
558 
559  return apduResponses;

562 void PoTransaction::processOpening(const AccessLevel& accessLevel)
563 {
564  mCurrentAccessLevel = accessLevel;
565 
566  /* create a sublist of AbstractPoCommandBuilder to be sent atomically */
567  std::vector<std::shared_ptr<AbstractPoCommandBuilder<
568  AbstractPoResponseParser>>> poAtomicCommands;
569 
570  std::atomic<int> neededSessionBufferSpace;
571  std::atomic<bool> overflow;
572 
573  for (const auto& commandBuilder : mPoCommandManager->getPoCommandBuilders())
574  {
575  /*
576  * Check if the command is a modifying one and get it status (overflow yes/no,
577  * neededSessionBufferSpace) if the command overflows the session buffer in atomic
578  * modification mode, an exception is raised.
579  */
580  if (checkModifyingCommand(commandBuilder, overflow, neededSessionBufferSpace)) {
581  if (overflow) {
582  /* Open the session with the current commands */
583  processAtomicOpening(mCurrentAccessLevel, poAtomicCommands);
584 
585  /*
586  * Closes the session, resets the modifications buffer counters for the next round

638 void PoTransaction::processPoCommandsInSession()
639 {
640  /* This method should be called only if a session was previously open */
641  checkSessionIsOpen();
642 
643  /* A session is open, we have to care about the PO modifications buffer */
644  std::vector<std::shared_ptr<AbstractPoCommandBuilder<AbstractPoResponseParser>>>
645  poAtomicBuilders;
646 
647  std::atomic<int> neededSessionBufferSpace;
648  std::atomic<bool> overflow;
649 
650  for (const auto& commandBuilder : mPoCommandManager->getPoCommandBuilders()) {
651  /*
652  * Check if the command is a modifying one and get it status (overflow yes/no,
653  * neededSessionBufferSpace) if the command overflows the session buffer in atomic
654  * modification mode, an exception is raised.
655  */
656  if (checkModifyingCommand(commandBuilder, overflow, neededSessionBufferSpace)) {
657  if (overflow) {
658  /*
659  * The current command would overflow the modifications buffer in the PO. We send
660  * the current commands and update the parsers. The parsers Iterator is kept all
661  * along the process.
662  */
663  processAtomicPoCommands(poAtomicBuilders, ChannelControl::KEEP_OPEN);
664 
665  /*
666  * Close the session and reset the modifications buffer counters for the next round
667  * (set the contact mode to avoid the transmission of the ratification)
668  */
669  processAtomicClosing(RatificationMode::CLOSE_RATIFIED, ChannelControl::KEEP_OPEN);
670  resetModificationsBufferCounter();
671 
672  /* We reopen a new session for the remaining commands to be sent */
673  processAtomicOpening(mCurrentAccessLevel);
674 
675  /*
676  * Clear the list and add the command that did not fit in the PO modifications
677  * buffer. We also update the usage counter without checking the result.
678  */
679  poAtomicBuilders.clear();
680  poAtomicBuilders.push_back(commandBuilder);
681 
682  /* Just update modifications buffer usage counter, ignore result (always false) */
683  isSessionBufferOverflowed(neededSessionBufferSpace);
684  } else {
685  /* The command fits in the PO modifications buffer, just add it to the list */
686  poAtomicBuilders.push_back(commandBuilder);
687  }
688  } else {
689  /* This command does not affect the PO modifications buffer */
690  poAtomicBuilders.push_back(commandBuilder);
691  }
692  }
693 
694  if (!poAtomicBuilders.empty())
695  processAtomicPoCommands(poAtomicBuilders, ChannelControl::KEEP_OPEN);
696 
697  /* Sets the flag indicating that the commands have been executed */
698  mPoCommandManager->notifyCommandsProcessed();
699 }

701 void PoTransaction::processClosing()
702 {
703  checkSessionIsOpen();
704 
705  bool atLeastOneReadCommand = false;
706  bool sessionPreviouslyClosed = false;
707 
708  std::atomic<int> neededSessionBufferSpace;
709  std::atomic<bool> overflow;
710 
711  std::vector<std::shared_ptr<AbstractPoCommandBuilder<AbstractPoResponseParser>>>
712  poAtomicCommands;
713 
714  for (const auto& commandBuilder : mPoCommandManager->getPoCommandBuilders()) {
715  /*
716  * Check if the command is a modifying one and get it status (overflow yes/no,
717  * neededSessionBufferSpace) if the command overflows the session buffer in atomic
718  * modification mode, an exception is raised.
719  */
720  if (checkModifyingCommand(commandBuilder, overflow, neededSessionBufferSpace)) {
721  if (overflow) {
722  /*
723  * Reopen a session with the same access level if it was previously closed in this
724  * current processClosing
725  */
726  if (sessionPreviouslyClosed)
727  processAtomicOpening(mCurrentAccessLevel);
728 
729  /*
730  * If at least one non-modifying was prepared, we use processAtomicPoCommands
731  * instead of processAtomicClosing to send the list
732  */
733  if (atLeastOneReadCommand) {
734  processAtomicPoCommands(poAtomicCommands, ChannelControl::KEEP_OPEN);
735 
736  /* Clear the list of commands sent */
737  poAtomicCommands.clear();
738  processAtomicClosing(poAtomicCommands,
739  RatificationMode::CLOSE_RATIFIED,
740  ChannelControl::KEEP_OPEN);
741  resetModificationsBufferCounter();
742  sessionPreviouslyClosed = true;
743  atLeastOneReadCommand = false;
744  } else {
745  /* All commands in the list are 'modifying the PO' */
746  processAtomicClosing(poAtomicCommands,
747  RatificationMode::CLOSE_RATIFIED,
748  ChannelControl::KEEP_OPEN);
749 
750  /* Clear the list of commands sent */
751  poAtomicCommands.clear();
752  resetModificationsBufferCounter();
753  sessionPreviouslyClosed = true;
754  }
755 
756  /*
757  * Add the command that did not fit in the PO modifications buffer. We also update
758  * the usage counter without checking the result.
759  */
760  poAtomicCommands.push_back(commandBuilder);
761 
762  /* Just update modifications buffer usage counter, ignore result (always false) */
763  isSessionBufferOverflowed(neededSessionBufferSpace);
764  } else {
765  /* The command fits in the PO modifications buffer, just add it to the list */
766  poAtomicCommands.push_back(commandBuilder);
767  }
768  } else {
769  /* This command does not affect the PO modifications buffer */
770  poAtomicCommands.push_back(commandBuilder);
771  atLeastOneReadCommand = true;
772  }
773  }
774 
775  if (sessionPreviouslyClosed)
776  /*
777  * Reopen if needed, to close the session with the requested conditions
778  * (CommunicationMode and channelControl)
779  */
780  processAtomicOpening(mCurrentAccessLevel);
781 
782  /* Finally, close the session as requested */
783  processAtomicClosing(poAtomicCommands,
784  mPoSecuritySettings->getRatificationMode(),
785  mChannelControl);
786 
787  /* Sets the flag indicating that the commands have been executed */
788  mPoCommandManager->notifyCommandsProcessed();
789 }
