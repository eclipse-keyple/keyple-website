34  std::atomic<bool>& cancellationFlag)
35 {
36  /*
37  * Loop until one the following condition is met : -
38  * AbstractObservableLocalReader#isSePresentPing returns false, meaning that the SE ping has
39  * failed - InterruptedException is caught
40  */
41 
42  long retries = 0;
43 
44  mLogger->debug("[%] Polling from isSePresentPing\n", mReader->getName());
45 
46  /* Re-init loop value to true */
47  mLoop = true;
48 
49  while (mLoop) {
50  if (cancellationFlag) {
51  mLogger->debug("[%] monitoring job cancelled\n",
52  mReader->getName());
53  return;
54  }
55 
56  if (!mReader->isSePresentPing()) {
57  mLogger->debug("[%] The SE stopped responding\n",
58  mReader->getName());
59  mLoop = false;
60  state->onEvent(InternalEvent::SE_REMOVED);
61  return;
62  }
63 
64  retries++;
65 
66  mLogger->trace("[%] Polling retries : %\n", mReader->getName(),
67  retries);
68 
69  try {
70  /* Wait for a bit */
71  Thread::sleep(mRemovalWait);
72  } catch (const InterruptedException& ignored) {
73  (void)ignored;
74  /* Restore interrupted state... */
75  mLoop = false;
76  std::terminate();
77  }
78  }

81 }
