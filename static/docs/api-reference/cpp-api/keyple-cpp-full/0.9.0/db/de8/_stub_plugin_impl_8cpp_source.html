56 void StubPluginImpl::plugStubReader(const std::string& readerName,
57  TransmissionMode transmissionMode,
58  bool synchronous)
59 {
60  mLogger->info("Plugging a new reader with readerName %\n", readerName);
61 
62  /* Add the native reader to the native readers list */
63  bool exist = mConnectedStubNames.find(readerName) != mConnectedStubNames.end();
64 
65  if (!exist && synchronous) {
66  /* add the reader as a new reader to the readers list */
67  auto reader = std::make_shared<StubReaderImpl>(getName(), readerName, transmissionMode);
68  mNativeReaders.insert({readerName, reader});
69  }
70 
71  mConnectedStubNames.insert(readerName);
72 
73  if (exist) {
74  mLogger->error("Reader with readerName % was already plugged\n", readerName);
75  }

78 void StubPluginImpl::plugStubReaders(const std::set<std::string>& readerNames,
79  bool synchronous)
80 {
81  mLogger->debug("Plugging % readers\n", readerNames.size());
82 
83  /* Plug stub readers that were not plugged already duplicate readerNames */
84  std::set<std::string> newNames = readerNames;
85 
86  /* Remove already connected stubNames */
87  for (auto& connected : mConnectedStubNames)
88  newNames.erase(connected);
89 
90  mLogger->debug("New readers to be created #%\n", newNames.size());
91 
92  /* Add new readerNames to the connectedStubNames */
93  if (newNames.size() > 0) {
94  if (synchronous) {
95  ConcurrentMap<const std::string, std::shared_ptr<StubReaderImpl>> newReaders;
96 
97  for (auto& name : newNames)
98  newReaders.insert({name, std::make_shared<StubReaderImpl>(this->getName(), name)});
99 
100  std::copy(newReaders.begin(),
101  newReaders.end(),
102  std::inserter(mNativeReaders, mNativeReaders.end()));
103  }
104 
105  std::copy(readerNames.begin(),
106  readerNames.end(),
107  std::inserter(mConnectedStubNames, mConnectedStubNames.end()));
108  } else {
109  mLogger->error("All % readers were already plugged\n", readerNames.size());

113 void StubPluginImpl::unplugStubReader(const std::string& readerName,
114  bool synchronous)
115 {
116  if (mConnectedStubNames.find(readerName) == mConnectedStubNames.end()) {
117  mLogger->warn("unplugStubReader() No reader found with name %\n", readerName);
118  } else {
119  /* Remove the reader from the readers list */
120  if (synchronous) {
121  mConnectedStubNames.erase(readerName);
122  mNativeReaders.erase(readerName);

124  mConnectedStubNames.erase(readerName);
125  }
126

128  mLogger->info("Unplugged reader with name %, connectedStubNames size : %\n",

134 void StubPluginImpl::unplugStubReaders(const std::set<std::string>& readerNames,
135  const bool synchronous)
136 {
137  mLogger->info("Unplug % stub readers\n", readerNames.size());
138  mLogger->debug("Unplug stub readers.. %\n", readerNames);
139 
140  std::list<std::shared_ptr<StubReaderImpl>> readersToDelete;
141 
142  for (const std::string name : readerNames) {
143  try {
144  readersToDelete.push_back(std::dynamic_pointer_cast<StubReaderImpl>(getReader(name)));
145  } catch (KeypleReaderNotFoundException& e) {
146  (void)e;
147  mLogger->warn("unplugStubReaders() No reader found with name %\n", name);
148  }
149  }
150 
151  for (const auto& readerName : readerNames)
152  mConnectedStubNames.erase(readerName);
153 
154  if (synchronous) {
155  for (const auto& reader : readersToDelete)
156  mNativeReaders.erase(reader->getName());
157  }
