243  thread[1] = std::thread(addReaderThread, std::ref(readers), 10, lock);
244  thread[2] = std::thread(removeReaderThread, std::ref(readers), 10, lock);

246  thread[4] = std::thread(addReaderThread, std::ref(readers), 10, lock);
247  thread[5] = std::thread(removeReaderThread, std::ref(readers), 10, lock);
248  thread[6] = std::thread(listReaders, std::ref(readers), 10, lock);
249  thread[7] = std::thread(removeReaderThread, std::ref(readers), 10, lock);
250  thread[8] = std::thread(listReaders, std::ref(readers), 10, lock);
251  thread[9] = std::thread(removeReaderThread, std::ref(readers), 10, lock);
