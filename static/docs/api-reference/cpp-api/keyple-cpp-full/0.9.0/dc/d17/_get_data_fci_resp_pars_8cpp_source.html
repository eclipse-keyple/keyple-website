95 GetDataFciRespPars::GetDataFciRespPars(
96  std::shared_ptr<ApduResponse> response, GetDataFciCmdBuild* builder)
97 : AbstractPoResponseParser(
98  response,
99  reinterpret_cast<AbstractPoCommandBuilder<AbstractPoResponseParser>*>(builder))
100 {
101  std::shared_ptr<TLV> tlv = nullptr;
102 
103  /* check the command status to determine if the DF has been invalidated */
104  if (response->getStatusCode() == 0x6283) {
105  logger->debug("The response to the select application command status"
106  "word indicates that the DF has been invalidated\n");
107  mIsDfInvalidated = true;
108  }
109 
110  /* parse the raw data with the help of the TLV class */
111  try {
112  /* init TLV object with the raw data and extract the FCI Template */
113  std::vector<uint8_t> responseData = response->getBytes();
114  tlv = std::make_shared<TLV>(responseData);
115 
116  /* Get the FCI template */
117  if (!tlv->parse(TAG_FCI_TEMPLATE, 0)) {
118  logger->error("FCI parsing error: FCI template tag not found\n");
119  return;
120  }
121 
122  /* Get the DF Name */
123  if (!tlv->parse(TAG_DF_NAME, tlv->getPosition())) {
124  logger->error("FCI parsing error: DF name tag not found\n");
125  return;
126  }
127 
128  mDfName = tlv->getValue();
129 
130  /* Get the FCI Proprietary Template */
131  if (!tlv->parse(TAG_FCI_PROPRIETARY_TEMPLATE, tlv->getPosition())) {
132  logger->error("FCI parsing error: FCI proprietary template tag "
133  "not found\n");
134  return;
135  }
136 
137  /* Get the FCI Issuer Discretionary Data */
138  if (!tlv->parse(TAG_FCI_ISSUER_DISCRETIONARY_DATA,
139  tlv->getPosition())) {
140  logger->error("FCI parsing error: FCI issuer discretionary data "
141  "tag not found\n");
142  return;
143  }
144 
145  /* Get the Application Serial Number */
146  if (!tlv->parse(TAG_APPLICATION_SERIAL_NUMBER, tlv->getPosition())) {
147  logger->error("FCI parsing error: serial number tag not found\n");
148  return;
149  }
150 
151  mApplicationSN = tlv->getValue();
152 
153  logger->debug("Application Serial Number = %\n",
154  ByteArrayUtil::toHex(mApplicationSN));
155 
156  /* Get the Discretionary Data */
157  if (!tlv->parse(TAG_DISCRETIONARY_DATA, tlv->getPosition())) {
158  logger->error("FCI parsing error: discretionary data tag not "
159  "found\n");
160  return;
161  }
162 
163  mDiscretionaryData = tlv->getValue();
164 
165  logger->debug("Discretionary Data = %\n", mDiscretionaryData);
166 
167  /* all 3 main fields were retrieved */
168  mIsValidCalypsoFCI = true;
169 
170  } catch (const std::runtime_error& e) {
171  /* Silently ignore problems decoding TLV structure. Just log. */
172  logger->debug("Error while parsing the FCI BER-TLV data structure "
173  "(%)\n", e.what());
174  }
175 }
