135  static int parseInt(const std::string& str, int radix, const bool decode)
136  {
137  if (!decode && str.empty())

139

142  bool isNeg = false;
143 
144  if (len == 0)
145  throw new NumberFormatException("string length is null");
146 
147  int ch = str.at(index);
148  if (ch == '-') {
149  if (len == 1)
150  throw new NumberFormatException("pure '-'");
151  isNeg = true;
152  ch = str.at(++index);
153  }
154 
155  if (decode) {
156  if (ch == '0') {
157  if (++index == len)
158  return 0;
159  if ((str.at(index) & ~('x' ^ 'X')) == 'X') {
160  radix = 16;
161  index++;
162  } else
163  radix = 8;
164  } else if (ch == '#') {
165  radix = 16;
166  index++;
167  }
168  }
169 
170  if (index == len)
171  throw new NumberFormatException("non terminated number: " + str);
172 
173  int max = MAX_VALUE / radix;
174  // We can't directly write `max = (MAX_VALUE + 1) / radix'.
175  // So instead we fake it.
176  if (isNeg && MAX_VALUE % radix == radix - 1)
177  ++max;
178 
179  int val = 0;
180  while (index < len) {
181  if (val < 0 || val > max)
182  throw NumberFormatException("number overflow (pos= " +
183  std::to_string(index) +

188  val = val * radix + ch;
