54 const std::vector<std::string>& PcscTerminal::listTerminals()
55 {
56  LONG ret;
57  SCARDCONTEXT context;
58  char* readers = NULL;
59  char* ptr = NULL;
60  DWORD len = 0;
61  static std::vector<std::string> list;
62 
63  /* Clear list */
64  list.clear();
65 
66  ret = SCardEstablishContext(SCARD_SCOPE_USER, NULL, NULL, &context);
67  if (ret != SCARD_S_SUCCESS) {
68  throw PcscTerminalException("SCardEstablishContext failed");
69  }
70 
71  ret = SCardListReaders(context, NULL, NULL, &len);
72  if (ret != SCARD_S_SUCCESS) {
73  throw PcscTerminalException("SCardListReaders failed");
74  }
75 
76  if (!len)
77  /* No readers to add to list */
78  return list;
79 
80  readers = (char*)calloc(len, sizeof(char));
81 
82  if (!readers)
83  /* No readers to add to list */
84  return list;
85 
86  ret = SCardListReaders(context, NULL, readers, &len);
87  if (ret != SCARD_S_SUCCESS) {
88  throw PcscTerminalException("SCardListReaders failed");
89  }
90 
91  ptr = readers;
92 
93  if (!ptr)
94  return list;
95 
96  while (*ptr) {
97  std::string s(ptr);
98  list.push_back(s);
99  ptr += strlen(ptr) + 1;
100  }
101 
102  SCardReleaseContext(context);
103  free(readers);
104 
105  return list;
106 }

168 void PcscTerminal::openAndConnect(const std::string& protocol)
169 {
170  LONG rv;
171  DWORD connectProtocol;
172  DWORD sharingMode = SCARD_SHARE_SHARED;
173  BYTE reader[200];
174  DWORD readerLen = sizeof(reader);
175  BYTE _atr[33];
176  DWORD atrLen = sizeof(_atr);
177 
178  mLogger->debug("[%] openAndConnect - protocol: %\n", mName, protocol);
179 
180  try {
181  establishContext();
182  } catch (PcscTerminalException& e) {
183  mLogger->error("openAndConnect - caught PcscTerminalException %\n", e);
184  throw;
185  }
186 
187  if (!protocol.compare("*")) {
188  connectProtocol = SCARD_PROTOCOL_T0 | SCARD_PROTOCOL_T1;
189  } else if (!protocol.compare("T=0")) {
190  connectProtocol = SCARD_PROTOCOL_T0;
191  } else if (!protocol.compare("T=1")) {
192  connectProtocol = SCARD_PROTOCOL_T1;
193  } else if (!protocol.compare("direct")) {
194  connectProtocol = 0;
195  sharingMode = SCARD_SHARE_DIRECT;
196  } else {
197  throw IllegalArgumentException("Unsupported protocol " + protocol);
198  }
199 
200  mLogger->debug("openAndConnect - connecting tp % with protocol: %, "
201  "connectProtocol: % and sharingMode: %\n",
202  mName,
203  protocol,
204  connectProtocol,
205  sharingMode);
206 
207  rv = SCardConnect(mContext, mName.c_str(), sharingMode,
208  connectProtocol, &mHandle, &mProtocol);
209  if (rv != SCARD_S_SUCCESS) {
210  mLogger->error("openAndConnect - SCardConnect failed (%)\n",
211  std::string(pcsc_stringify_error(rv)));
212  releaseContext();
213  throw PcscTerminalException("openAndConnect failed");
214  }
215 
216  switch (mProtocol) {
217  case SCARD_PROTOCOL_T0:
218  mPioSendPCI = *SCARD_PCI_T0;
219  break;
220  case SCARD_PROTOCOL_T1:
221  mPioSendPCI = *SCARD_PCI_T1;
222  break;
223  }
224 
225  rv = SCardStatus(mHandle, (LPSTR)reader, &readerLen, &mState,
226  &mProtocol, _atr, &atrLen);
227  if (rv != SCARD_S_SUCCESS) {
228  mLogger->error("openAndConnect - SCardStatus failed (s)\n",
229  std::string(pcsc_stringify_error(rv)));
230  releaseContext();
231  throw PcscTerminalException("openAndConnect failed");
232  } else {
233  mLogger->debug("openAndConnect - card state: %\n", mState);
234  }
235 
236  mAtr.clear();
237  mAtr.insert(mAtr.end(), _atr, _atr + atrLen);
238 }

254 std::vector<uint8_t> PcscTerminal::transmitApdu(const std::vector<uint8_t>& apduIn)
255 {
256  if (apduIn.size() == 0)
257  throw IllegalArgumentException("command cannot be empty");
258 
259  /* Make a copy */
260  std::vector<uint8_t> _apduIn = apduIn;
261 
262  /* To check */
263  bool t0GetResponse = true;
264  bool t1GetResponse = true;
265  bool t1StripLe = true;
266 
267  /*
268  * Note that we modify the 'command' array in some cases, so it must be a copy of the
269  * application provided data
270  */
271  int n = _apduIn.size();
272  bool t0 = mProtocol == SCARD_PROTOCOL_T0;
273  bool t1 = mProtocol == SCARD_PROTOCOL_T1;
274 
275  if (t0 && (n >= 7) && (_apduIn[4] == 0))
276  throw PcscTerminalException("Extended len. not supported for T=0");
277 
278  if ((t0 || (t1 && t1StripLe)) && (n >= 7)) {
279  int lc = _apduIn[4] & 0xff;
280  if (lc != 0) {
281  if (n == lc + 6) {
282  n--;
283  }
284  } else {
285  lc = ((_apduIn[5] & 0xff) << 8) | (_apduIn[6] & 0xff);
286  if (n == lc + 9) {
287  n -= 2;
288  }
289  }
290  }
291 
292  bool getresponse = (t0 && t0GetResponse) || (t1 && t1GetResponse);
293  int k = 0;
294  std::vector<uint8_t> result;
295 
296  while (true) {
297  if (++k >= 32) {
298  throw PcscTerminalException("Could not obtain response");
299  }
300 
301  char r_apdu[261];
302  DWORD dwRecv = sizeof(r_apdu);
303  long rv;
304 
305  mLogger->debug("[%] transmitApdu - c-apdu >> %\n", mName, _apduIn);
306 
307  rv = SCardTransmit(mHandle,
308  &mPioSendPCI,
309  (LPCBYTE)_apduIn.data(),
310  _apduIn.size(),
311  NULL,
312  (LPBYTE)r_apdu,
313  &dwRecv);
314  if (rv != SCARD_S_SUCCESS) {
315  mLogger->error("SCardTransmit failed with error: %\n",
316  std::string(pcsc_stringify_error(rv)));
317  throw PcscTerminalException("ScardTransmit failed");
318  }
319 
320  std::vector<uint8_t> response(r_apdu, r_apdu + dwRecv);
321 
322  mLogger->debug("[%] transmitApdu - r-apdu << %\n", mName, response);
323 
324  int rn = response.size();
325  if (getresponse && (rn >= 2)) {
326  /* See ISO 7816/2005, 5.1.3 */
327  if ((rn == 2) && (response[0] == 0x6c)) {
328  // Resend command using SW2 as short Le field
329  _apduIn[n - 1] = response[1];
330  continue;
331  }
332 
333  if (response[rn - 2] == 0x61) {
334  /* Issue a GET RESPONSE command with the same CLA using SW2 as short Le field */
335  if (rn > 2)
336  result.insert(result.end(), response.begin(), response.begin() + rn - 2);
337 
338  std::vector<uint8_t> getResponse(5);
339  getResponse[0] = _apduIn[0];
340  getResponse[1] = 0xC0;
341  getResponse[2] = 0;
342  getResponse[3] = 0;
343  getResponse[4] = response[rn - 1];
344  n = 5;
345  _apduIn.swap(getResponse);
346  continue;
347  }
348  }
349 
350  result.insert(result.end(), response.begin(), response.begin() + rn);
351  break;
352  }
353 
354  return result;
355 }
