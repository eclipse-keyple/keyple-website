92 std::shared_ptr<ReaderEvent> AbstractObservableLocalReader::processSeInserted()
93 {
94  mLogger->trace("[%] process the inserted se\n", getName());
95 
96  if (mDefaultSelectionsRequest == nullptr) {
97  mLogger->trace("[%] no default selection request defined, notify " \
98  "SE_INSERTED\n", getName());
99 
100  /* No default request is defined, just notify the SE insertion */
101  return std::make_shared<ReaderEvent>(
102  getPluginName(), getName(), ReaderEvent::EventType::SE_INSERTED,
103  nullptr);
104  } else {
105  /*
106  * A default request is defined, send it and notify according to the
107  * notification mode and the selection status
108  */
109  try {
110  std::vector<std::shared_ptr<SeResponse>> seResponses =
111  transmitSeRequests(
112  mDefaultSelectionsRequest->getSelectionSeRequests(),
113  mDefaultSelectionsRequest->getMultiSeRequestProcessing(),
114  mDefaultSelectionsRequest->getChannelControl());
115 
116  bool aSeMatched = false;
117 
118  for (auto seResponse : seResponses) {
119  if (seResponse != nullptr && seResponse->getSelectionStatus()->hasMatched()) {
120  mLogger->trace("[%] a default selection has matched\n", getName());
121  aSeMatched = true;
122  break;
123  }
124  }
125 
126  if (mNotificationMode ==
127  ObservableReader::NotificationMode::MATCHED_ONLY) {
128  /* Notify only if a SE matched the selection, just ignore if not */
129  if (aSeMatched) {
130  return std::make_shared<ReaderEvent>(
131  getPluginName(),
132  getName(),
133  ReaderEvent::EventType::SE_MATCHED,
134  std::make_shared<DefaultSelectionsResponse>(seResponses));
135  } else {
136  mLogger->trace("[%] selection hasn't matched, do not throw" \
137  " any event because of MATCHED_ONLY flag\n",
138  getName());
139  return nullptr;
140  }
141  } else {
142  /* ObservableReader::NotificationMode::ALWAYS */
143  if (aSeMatched) {
144  /* The SE matched, notify a SE_MATCHED event with the received response */
145  return std::make_shared<ReaderEvent>(
146  getPluginName(),
147  getName(),
148  ReaderEvent::EventType::SE_MATCHED,
149  std::make_shared<DefaultSelectionsResponse>(seResponses));
150  } else {
151  /*
152  * The SE didn't match, notify an SE_INSERTED event with the
153  * received response
154  */
155  mLogger->trace("[%] none of % default selection matched\n",
156  getName(),
157  seResponses.size());
158 
159  return std::make_shared<ReaderEvent>(
160  getPluginName(),
161  getName(),
162  ReaderEvent::EventType::SE_INSERTED,
163  std::make_shared<DefaultSelectionsResponse>(seResponses));
164  }
165  }
166  } catch (const KeypleReaderException& e) {
167  /* The last transmission failed, close the logical and physical channels */
168  closeLogicalAndPhysicalChannels();
169  mLogger->debug("An IO Exception occurred while processing the "
170  "default selection. %\n", e);
171  /*
172  * In this case the SE has been removed or not read correctly, do
173  * not throw event
174  */
175  }
176  }
177 
178  /*
179  * We close here the physical channel in case it has been opened for a
180  * SE outside the expected SEs
181  */
182  try {
183  closePhysicalChannel();
184  } catch (const KeypleReaderIOException& e) {
185  mLogger->error("Error while closing physical channel. %\n", e);
186  }
187 
188  /* No event returned */
189  return nullptr;
190 }

268 void AbstractObservableLocalReader::notifyObservers(const std::shared_ptr<ReaderEvent> event)
269 {
270  mLogger->trace("[%] Notifying a reader event to % observers. EVENTNAME = %\n",
271  getName(),
272  countObservers(),
273  event->getEventType().getName());
274 
275  if (mShuttingDown)
276  return;
277 
278  std::lock_guard<std::recursive_mutex> lock(mMutex);
279  for (auto& observer : mObservers) {
280  if (mObservers.size() != 0) {
281  mLogger->trace("[%] Calling update on observer\n", getName());
