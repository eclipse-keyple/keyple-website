62 std::shared_ptr<SelectionsResult> SeSelection::processSelection(
63  std::shared_ptr<AbstractDefaultSelectionsResponse> defaultSelectionsResponse)
64 {
65  int index = 0;
66 
67  std::shared_ptr<SelectionsResult> selectionsResult =
68  std::make_shared<SelectionsResult>();
69 
70  /* Check SeResponses */
71  for (const auto& seResponse :
72  (std::dynamic_pointer_cast<DefaultSelectionsResponse>(
73  defaultSelectionsResponse))
74  ->getSelectionSeResponses()) {
75  /*
76  * Test if the selection is successful: we should have either a FCI
77  * or an ATR
78  */
79  if (seResponse != nullptr &&
80  seResponse->getSelectionStatus() != nullptr &&
81  seResponse->getSelectionStatus()->hasMatched()) {
82  /*
83  * Create a AbstractMatchingSe with the class deduced from the
84  * selection request during the selection preparation
85  */
86  const std::shared_ptr<AbstractMatchingSe> matchingSe =
87  mSeSelectionRequests[index]->parse(seResponse);
88 
89  /* Determine if the current matching SE is selected */
90  std::shared_ptr<SelectionStatus> selectionStatus =
91  seResponse->getSelectionStatus();
92  bool isSelected;
93  if (selectionStatus)
94  isSelected = selectionStatus->hasMatched() &&
95  seResponse->isLogicalChannelOpen();
96  else
97  isSelected = false;
98 
99  selectionsResult->addMatchingSe(index, matchingSe, isSelected);
100  }
101 
102  index++;
103  }
104 
105  return selectionsResult;
106 }

125 std::shared_ptr<SelectionsResult> SeSelection::processExplicitSelection(
126  std::shared_ptr<SeReader> seReader)
127 {
128  std::vector<std::shared_ptr<SeRequest>> selectionRequests;
129 
130  std::for_each(mSeSelectionRequests.begin(),
131  mSeSelectionRequests.end(),
132  [&](const std::shared_ptr<AbstractSeSelectionRequest<
133  AbstractApduCommandBuilder>>& r) {
134  selectionRequests.push_back(r->getSelectionRequest());
135  });
136 
137  logger->trace("Transmit SELECTIONREQUEST (% request(s))\n",
138  selectionRequests.size());
139 
140  if (!seReader) {
141  logger->error("processExplicitSelection - seReader is null\n");
142  return nullptr;
143  }
144 
145  /* Communicate with the SE to do the selection */
146  auto proxyReader = std::dynamic_pointer_cast<ProxyReader>(seReader);
147  std::vector<std::shared_ptr<SeResponse>> seResponses =
148  proxyReader->transmitSeRequests(selectionRequests,
149  mMultiSeRequestProcessing,
150  mChannelControl);
151 
152  return processSelection(
153  std::make_shared<DefaultSelectionsResponse>(seResponses));
