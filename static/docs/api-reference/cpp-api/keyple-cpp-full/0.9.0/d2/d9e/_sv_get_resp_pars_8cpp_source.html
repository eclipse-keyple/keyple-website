78 SvGetRespPars::SvGetRespPars(const std::vector<uint8_t>& svCommandHeader,
79  std::shared_ptr<ApduResponse> response,
80  SvGetCmdBuild* builder)
81 : AbstractPoResponseParser(
82  response,
83  reinterpret_cast<AbstractPoCommandBuilder<AbstractPoResponseParser>*>(builder)),
84  mSvCommandHeader(svCommandHeader)
85 {
86  const std::vector<uint8_t>& poResponse = response->getDataOut();
87 
88  switch (poResponse.size()) {
89  case 0x21: /* Compatibility mode, Reload */
90  case 0x1E: /* Compatibility mode, Debit or Undebit */
91  mChallengeOut = std::vector<uint8_t>(2);
92  mPreviousSignatureLo = std::vector<uint8_t>(3);
93  mCurrentKVC = poResponse[0];
94  mTransactionNumber = ByteArrayUtil::twoBytesToInt(poResponse, 1);
95  System::arraycopy(poResponse, 3, mPreviousSignatureLo, 0, 3);
96  mChallengeOut[0] = poResponse[6];
97  mChallengeOut[1] = poResponse[7];
98  mBalance = ByteArrayUtil::threeBytesSignedToInt(poResponse, 8);
99  if (poResponse.size() == 0x21) {
100  /* Reload */
101  mLoadLog = std::make_shared<SvLoadLogRecord>(poResponse, 11);
102  mDebitLog = nullptr;
103  } else {
104  /* Debit */
105  mLoadLog = nullptr;
106  mDebitLog = std::make_shared<SvDebitLogRecord>(poResponse, 11);
107  }
108  break;
109  case 0x3D: /* Revision 3.2 mode */
110  mChallengeOut = std::vector<uint8_t>(8);
111  mPreviousSignatureLo = std::vector<uint8_t>(6);
112  System::arraycopy(poResponse, 0, mChallengeOut, 0, 8);
113  mCurrentKVC = poResponse[8];
114  mTransactionNumber = ByteArrayUtil::twoBytesToInt(poResponse, 9);
115  System::arraycopy(poResponse, 11, mPreviousSignatureLo, 0, 6);
116  mBalance = ByteArrayUtil::threeBytesSignedToInt(poResponse, 17);
117  mLoadLog = std::make_shared<SvLoadLogRecord>(poResponse, 20);
118  mDebitLog = std::make_shared<SvDebitLogRecord>(poResponse, 42);
119  break;
120  default:
121  throw IllegalStateException("Incorrect data length in response to SVGet");
122  }
