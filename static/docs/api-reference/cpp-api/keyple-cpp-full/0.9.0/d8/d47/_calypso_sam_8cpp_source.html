44 CalypsoSam::CalypsoSam(std::shared_ptr<SeResponse> selectionResponse,
45  const TransmissionMode& transmissionMode)
46 : AbstractMatchingSe(selectionResponse, transmissionMode),
47  mSamRevision(SamRevision::C1), /* Default value to please compiler */
48  mSerialNumber(4)
49 {
50  std::string atrString = ByteArrayUtil::toHex(
51  selectionResponse->getSelectionStatus()->getAtr()->getBytes());
52  if (atrString.empty()) {
53  throw IllegalStateException("ATR should not be empty.");
54  }
55 
56  /* extract the historical bytes from T3 to T12 */
57  std::string extractRegex = "3B(.{6}|.{10})805A(.{20})829000";
58  std::unique_ptr<Pattern> pattern = Pattern::compile(extractRegex);
59  std::unique_ptr<Matcher> matcher = pattern->matcher(atrString);
60  if (matcher->find(0)) {
61  std::vector<uint8_t> atrSubElements =
62  ByteArrayUtil::fromHex(matcher->group(2));
63  mPlatform = atrSubElements[0];
64  mApplicationType = atrSubElements[1];
65  mApplicationSubType = atrSubElements[2];
66 
67  // determine SAM revision from Application Subtype
68  switch (mApplicationSubType) {
69  case 0xC1:
70  mSamRevision = SamRevision(SamRevision::C1);
71  break;
72  case 0xD0:
73  case 0xD1:
74  case 0xD2:
75  mSamRevision = SamRevision(SamRevision::S1D);
76  break;
77  case 0xE1:
78  mSamRevision = SamRevision(SamRevision::S1E);
79  break;
80  default:
81  throw IllegalStateException(
82  "Unknown SAM revision (unrecognized application subtype " +
83  StringHelper::uint8ToHexString(mApplicationSubType));
84  }
85 
86  mSoftwareIssuer = atrSubElements[3];
87  mSoftwareVersion = atrSubElements[4];
88  mSoftwareRevision = atrSubElements[5];
89  System::arraycopy(atrSubElements, 6, mSerialNumber, 0, 4);
90  mLogger->trace("SAM % PLATFORM = %, APPTYPE = %, APPSUBTYPE = %, " \
91  "SWISSUER = %, SWVERSION = %, SWREVISION = %\n",
92  mSamRevision.getName(),
93  mPlatform,
94  mApplicationType,
95  mApplicationSubType,
96  mSoftwareIssuer,
97  mSoftwareVersion,
98  mSoftwareRevision);
99 
100  mLogger->trace("SAM SERIALNUMBER = %\n", mSerialNumber);
101  } else {
102  throw IllegalStateException("Unrecognized ATR structure: " + atrString);
103  }
104 }
