63 const std::shared_ptr<StatusProperties> AbstractApduResponseParser::getStatusCodeProperties() const
64 {
65  int sc = mResponse->getStatusCode();
66  const std::map<int, std::shared_ptr<StatusProperties>>& st = getStatusTable();
67 
68  /*
69  * /!\ Cannot use operator[] to check key/value existence since an unknown

73  if ((it = st.find(sc)) != st.end())
74  return it->second;
75  else
76  return nullptr;

93 void AbstractApduResponseParser::checkStatus() const
94 {
95  std::shared_ptr<StatusProperties> props = getStatusCodeProperties();
96 
97  if (props != nullptr && props->isSuccessful())
98  return;
99 
100  /* Status code is not referenced, or not successful */
101 
102  /* Exception class */
103  const std::type_info& exceptionClass =
104  props != nullptr ? props->getExceptionClass() : typeid(Exception);
105 
106  /* Message */
107  const std::string& message =
108  props != nullptr ? props->getInformation() : "Unknown status";
109 
110  /* Command reference */
111  std::shared_ptr<SeCommand> commandRef = getCommandRef();
112 
113  /* status code */
114  int statusCode = mResponse->getStatusCode();
115 
116  /* Throw the exception */
117  throw buildCommandException(exceptionClass, message, commandRef, statusCode);
118 }
