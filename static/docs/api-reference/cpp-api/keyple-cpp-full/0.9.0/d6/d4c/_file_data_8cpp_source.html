137 void FileData::setContent(const int numRecord,
138  const std::vector<uint8_t>& content,
139  const int offset)
140 {
141  std::vector<uint8_t> newContent;
142  std::vector<uint8_t> oldContent;
143  const int newLength = offset + content.size();
144 
145  const auto it = mRecords.find(numRecord);
146  if (it != mRecords.end())
147  oldContent = it->second;
148 
149  if (static_cast<int>(oldContent.size()) == 0) {
150  newContent = std::vector<uint8_t>(newLength);
151  } else if (static_cast<int>(oldContent.size()) <= offset) {
152  newContent = std::vector<uint8_t>(newLength);
153  System::arraycopy(oldContent, 0, newContent, 0, oldContent.size());
154  } else if (static_cast<int>(oldContent.size()) < newLength) {
155  newContent = std::vector<uint8_t>(newLength);
156  System::arraycopy(oldContent, 0, newContent, 0, offset);
157  } else {
158  newContent = oldContent;
159  }
160 
161  System::arraycopy(content, 0, newContent, offset, content.size());
162 
163  mRecords[numRecord] = newContent;
164 }

166 void FileData::fillContent(const int numRecord, std::vector<uint8_t>& content)
167 {
168  /*
169  * /!\ Make sure 'actualContent' *is* the vector from the map and not a copy
170  */
171  const auto it = mRecords.find(numRecord);
172  if (it == mRecords.end()) {
173  mRecords.insert({numRecord, content});
174  } else {
175  std::vector<uint8_t>& actualContent = it->second;
176 
177  if (static_cast<int>(actualContent.size()) <
178  static_cast<int>(content.size())) {
179  for (int i = 0; i < static_cast<int>(actualContent.size()); i++)
180  content[i] |= actualContent[i];
181  mRecords[numRecord] = content;
182  } else {
183  for (int i = 0; i < static_cast<int>(content.size()); i++)
184  actualContent[i] |= content[i];
185  }
186  }
