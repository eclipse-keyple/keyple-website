193  "Unknown file type: " +

379 std::unique_ptr<DirectoryHeader> CalypsoPoUtils::createDirectoryHeader(
380  const std::vector<uint8_t>& proprietaryInformation)
381 {
382  std::vector<uint8_t> lAccessConditions(SEL_AC_LENGTH);
383  System::arraycopy(proprietaryInformation, SEL_AC_OFFSET, lAccessConditions, 0, SEL_AC_LENGTH);
384 
385  std::vector<uint8_t> lKeyIndexes(SEL_NKEY_LENGTH);
386  System::arraycopy(proprietaryInformation, SEL_NKEY_OFFSET, lKeyIndexes, 0, SEL_NKEY_LENGTH);
387 
388  const uint8_t lDfStatus = proprietaryInformation[SEL_DF_STATUS_OFFSET];
389 
390  const uint16_t lLid =
391  (((proprietaryInformation[SEL_LID_OFFSET] << 8) & 0xff00) |
392  (proprietaryInformation[SEL_LID_OFFSET + 1] & 0x00ff));
393 
394  return DirectoryHeader::builder()
395  ->lid(lLid)
396  .accessConditions(lAccessConditions)
397  .keyIndexes(lKeyIndexes)
398  .dfStatus(lDfStatus)
399  .kvc(PoTransaction::SessionSetting::AccessLevel::SESSION_LVL_PERSO,
400  proprietaryInformation[SEL_KVCS_OFFSET])
401  .kvc(PoTransaction::SessionSetting::AccessLevel::SESSION_LVL_LOAD,
402  proprietaryInformation[SEL_KVCS_OFFSET + 1])
403  .kvc(PoTransaction::SessionSetting::AccessLevel::SESSION_LVL_DEBIT,
404  proprietaryInformation[SEL_KVCS_OFFSET + 2])
405  .kif(PoTransaction::SessionSetting::AccessLevel::SESSION_LVL_PERSO,
406  proprietaryInformation[SEL_KIFS_OFFSET])
407  .kif(PoTransaction::SessionSetting::AccessLevel::SESSION_LVL_LOAD,
408  proprietaryInformation[SEL_KIFS_OFFSET + 1])
409  .kif(PoTransaction::SessionSetting::AccessLevel::SESSION_LVL_DEBIT,
410  proprietaryInformation[SEL_KIFS_OFFSET + 2])
411  .build();
412 }

442 std::unique_ptr<FileHeader> CalypsoPoUtils::createFileHeader(
443  const std::vector<uint8_t>& proprietaryInformation)
444 {
445  const FileType fileType = getEfTypeFromPoValue(proprietaryInformation[SEL_EF_TYPE_OFFSET]);
446 
447  int lRecordSize;
448  int lRecordsNumber;
449 
450  if (fileType == FileType::BINARY) {
451  lRecordSize = ((proprietaryInformation[SEL_REC_SIZE_OFFSET] << 8) & 0x0000ff00) |
452  (proprietaryInformation[SEL_NUM_REC_OFFSET] & 0x000000ff);
453  lRecordsNumber = 1;
454  } else {
455  lRecordSize = proprietaryInformation[SEL_REC_SIZE_OFFSET];
456  lRecordsNumber = proprietaryInformation[SEL_NUM_REC_OFFSET];
457  }
458 
459  std::vector<uint8_t> lAccessConditions(SEL_AC_LENGTH);
460  System::arraycopy(proprietaryInformation, SEL_AC_OFFSET, lAccessConditions, 0, SEL_AC_LENGTH);
461 
462  std::vector<uint8_t> lKeyIndexes(SEL_NKEY_LENGTH);
463  System::arraycopy(proprietaryInformation, SEL_NKEY_OFFSET, lKeyIndexes, 0, SEL_NKEY_LENGTH);
464 
465  const uint8_t lDfStatus = proprietaryInformation[SEL_DF_STATUS_OFFSET];
466 
467  const uint16_t lSharedReference =
468  (((proprietaryInformation[SEL_DATA_REF_OFFSET] << 8) & 0xff00) |
469  (proprietaryInformation[SEL_DATA_REF_OFFSET + 1] & 0x00ff));
470 
471  const uint16_t lLid =
472  (((proprietaryInformation[SEL_LID_OFFSET] << 8) & 0xff00) |
473  (proprietaryInformation[SEL_LID_OFFSET + 1] & 0x00ff));
474 
475  return FileHeader::builder()
476  ->lid(lLid)
477  .recordsNumber(lRecordsNumber)
478  .recordSize(lRecordSize)
479  .type(fileType)
480  .accessConditions(lAccessConditions)
481  .keyIndexes(lKeyIndexes)
482  .dfStatus(lDfStatus)
483  .sharedReference(lSharedReference)
484  .build();
485 }

487 std::shared_ptr<AbstractPoResponseParser> CalypsoPoUtils::updateCalypsoPo(
488  std::shared_ptr<CalypsoPo> calypsoPo,
489  std::shared_ptr<AbstractPoCommandBuilder<AbstractPoResponseParser>> commandBuilder,
490  std::shared_ptr<ApduResponse> apduResponse)
491 {
492  const uint8_t ins =
493  commandBuilder->getCommandRef()->getInstructionByte();
494 
495  if (ins == CalypsoPoCommand::READ_RECORDS.getInstructionByte())
496  return updateCalypsoPoReadRecords(
497  calypsoPo,
498  std::reinterpret_pointer_cast<ReadRecordsCmdBuild>(commandBuilder),
499  apduResponse);
500  else if (ins == CalypsoPoCommand::SELECT_FILE.getInstructionByte())
501  return updateCalypsoPoSelectFile(
502  calypsoPo,
503  std::reinterpret_pointer_cast<SelectFileCmdBuild>(commandBuilder),
504  apduResponse);
505  else if (ins == CalypsoPoCommand::UPDATE_RECORD.getInstructionByte())
506  return updateCalypsoPoUpdateRecord(
507  calypsoPo,
508  std::reinterpret_pointer_cast<UpdateRecordCmdBuild>(commandBuilder),
509  apduResponse);
510  else if (ins == CalypsoPoCommand::WRITE_RECORD.getInstructionByte())
511  return updateCalypsoPoWriteRecord(
512  calypsoPo,
513  std::reinterpret_pointer_cast<WriteRecordCmdBuild>(commandBuilder),
514  apduResponse);
515  else if (ins == CalypsoPoCommand::APPEND_RECORD.getInstructionByte())
516  return updateCalypsoPoAppendRecord(
517  calypsoPo,
518  std::reinterpret_pointer_cast<AppendRecordCmdBuild>(commandBuilder),
519  apduResponse);
520  else if (ins == CalypsoPoCommand::DECREASE.getInstructionByte())
521  return updateCalypsoPoDecrease(
522  calypsoPo,
523  std::reinterpret_pointer_cast<DecreaseCmdBuild>(commandBuilder),
524  apduResponse);
525  else if (ins == CalypsoPoCommand::INCREASE.getInstructionByte())
526  return updateCalypsoPoIncrease(
527  calypsoPo,
528  std::reinterpret_pointer_cast<IncreaseCmdBuild>(commandBuilder),
529  apduResponse);
530  else if (ins == CalypsoPoCommand::OPEN_SESSION_10.getInstructionByte() ||
531  ins == CalypsoPoCommand::OPEN_SESSION_24.getInstructionByte() ||
532  ins == CalypsoPoCommand::OPEN_SESSION_31.getInstructionByte() ||
533  ins == CalypsoPoCommand::OPEN_SESSION_32.getInstructionByte())
534  return updateCalypsoPoOpenSession(
535  calypsoPo,
536  std::reinterpret_pointer_cast<
537  AbstractOpenSessionCmdBuild<AbstractOpenSessionRespPars>>(commandBuilder),
538  apduResponse);
539  else if (ins == CalypsoPoCommand::CLOSE_SESSION.getInstructionByte())
540  return updateCalypsoPoCloseSession(
541  calypsoPo,
542  std::reinterpret_pointer_cast<CloseSessionCmdBuild>(commandBuilder),
543  apduResponse);
544  else if (ins == CalypsoPoCommand::GET_CHALLENGE.getInstructionByte())
545  return updateCalypsoPoGetChallenge(
546  std::reinterpret_pointer_cast<PoGetChallengeCmdBuild>(commandBuilder),
547  apduResponse);
548  else if (ins == CalypsoPoCommand::VERIFY_PIN.getInstructionByte())
549  return updateCalypsoVerifyPin(
550  calypsoPo,
551  std::reinterpret_pointer_cast<VerifyPinCmdBuild>(commandBuilder),
552  apduResponse);
553  else if (ins == CalypsoPoCommand::SV_GET.getInstructionByte())
554  return updateCalypsoPoSvGet(
555  calypsoPo,
556  std::reinterpret_pointer_cast<SvGetCmdBuild>(commandBuilder),
557  apduResponse);
558  else if (ins == CalypsoPoCommand::SV_RELOAD.getInstructionByte() ||
559  ins == CalypsoPoCommand::SV_DEBIT.getInstructionByte() ||
560  ins == CalypsoPoCommand::SV_UNDEBIT.getInstructionByte())
561  return updateCalypsoPoSvOperation(
562  calypsoPo,
563  std::reinterpret_pointer_cast
564  <AbstractPoCommandBuilder<AbstractPoResponseParser>>(commandBuilder),
565  apduResponse);
566  else if (ins == CalypsoPoCommand::INVALIDATE.getInstructionByte() ||
567  ins == CalypsoPoCommand::REHABILITATE.getInstructionByte())
568  return updateCalypsoInvalidateRehabilitate(
569  calypsoPo,
570  std::reinterpret_pointer_cast
571  <AbstractPoCommandBuilder<AbstractPoResponseParser>>(commandBuilder),
572  apduResponse);
573  else if (ins == CalypsoPoCommand::CHANGE_KEY.getInstructionByte() ||
