33 SvDebitCmdBuild::SvDebitCmdBuild(const PoClass& poClass,

39 : AbstractPoCommandBuilder<SvDebitRespPars>(
40  std::make_shared<CalypsoPoCommand>(CalypsoPoCommand::SV_DEBIT), nullptr),
41  mPoClass(poClass),
42  mPoRevision(poRevision)
43 {
44  /* @see Calypso Layer ID 8.02 (200108) */
45  if (amount < 0 || amount > 32767)
46  throw IllegalArgumentException("Amount is outside allowed boundaries (0 <= amount <= " \
47  "32767)");
48 
49  if (date.empty() || time.empty())
50  throw IllegalArgumentException("date and time cannot be null");
51 
52  if (date.size() != 2 || time.size() != 2)
53  throw IllegalArgumentException("date and time must be 2-byte arrays");
54 
55  /*
56  * Handle the dataIn size with signatureHi length according to PO revision (3.2 rev have a
57  * 10-byte signature)
58  */
59  mDataIn = std::vector<uint8_t>(15 + (poRevision == PoRevision::REV3_2 ? 10 : 5));
60 
61  /* dataIn[0] will be filled in at the finalization phase */
62  uint16_t shortAmount = static_cast<uint16_t>(-amount);
63  mDataIn[1] = ((shortAmount >> 8) & 0xFF);
64  mDataIn[2] = (shortAmount & 0xFF);
65  mDataIn[3] = date[0];
66  mDataIn[4] = date[1];
67  mDataIn[5] = time[0];
68  mDataIn[6] = time[1];
69  mDataIn[7] = kvc;
70  // dataIn[8]..dataIn[8+7+sigLen] will be filled in at the finalization phase.
71 }

73 void SvDebitCmdBuild::finalizeBuilder(const std::vector<uint8_t>& debitComplementaryData)
74 {
75  if ((mPoRevision == PoRevision::REV3_2 && debitComplementaryData.size() != 20) ||
76  (mPoRevision != PoRevision::REV3_2 && debitComplementaryData.size() != 15))
77  throw IllegalArgumentException("Bad SV prepare load data length.");
78 
79  const uint8_t p1 = debitComplementaryData[4];
80  const uint8_t p2 = debitComplementaryData[5];
81 
82  mDataIn[0] = debitComplementaryData[6];
83  System::arraycopy(debitComplementaryData, 0, mDataIn, 8, 4);
84  System::arraycopy(debitComplementaryData, 7, mDataIn, 12, 3);
85  System::arraycopy(debitComplementaryData, 10, mDataIn, 15, debitComplementaryData.size() - 10);
86 
87  mRequest = setApduRequest(mPoClass.getValue(), mCommand, p1, p2, mDataIn);
