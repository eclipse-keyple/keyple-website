53 SamResourceManagerDefault::SamResourceManagerDefault(std::shared_ptr<ReaderPlugin> readerPlugin,
54  const std::string& samReaderFilter,
55  const int maxBlockingTime,
56  const int sleepTime)
57 : mSamReaderPlugin(readerPlugin), mMaxBlockingTime(maxBlockingTime), mSleepTime(sleepTime)
58 {
59  /* Assign parameters */
60  if (sleepTime < 1)
61  throw IllegalArgumentException("Sleep time must be greater than 0");
62 
63  if (maxBlockingTime < 1)
64  throw IllegalArgumentException("Max Blocking Time must be greater than 0");
65 
66  mReaderObserver = std::make_shared<ReaderObserver>(*this);
67 
68  mLogger->info("PLUGINNAME = % initialize the localManagedSamResources " \
69  "with the % connected readers filtered by %\n",
70  mSamReaderPlugin->getName(),
71  mSamReaderPlugin->getReaders().size(),
72  samReaderFilter);
73 
74  std::unique_ptr<Pattern> p = Pattern::compile(samReaderFilter);
75 
76  const ConcurrentMap<const std::string, std::shared_ptr<SeReader>>& readers =
77  mSamReaderPlugin->getReaders();
78 
79  for (const auto& samReader : readers) {
80  if (p->matcher(samReader.first)->matches()) {
81  mLogger->trace("Add reader: %\n", samReader.first);
82  try {
83  initSamReader(mSamReaderPlugin->getReader(samReader.first), mReaderObserver);
84  } catch (const KeypleReaderException& e) {
85  mLogger->error("could not init samReader %\n", samReader.second, e);
86  }
87  } else {
88  mLogger->trace("Reader not matching: %\n", samReader.second);
89  }
90  }
91 
92  if (std::dynamic_pointer_cast<ObservablePlugin>(readerPlugin)) {
93  /* Add an observer to monitor reader and SAM insertions */
94  auto pluginObserver = std::make_shared<PluginObserver>(mReaderObserver,
95  samReaderFilter,
96  *this);
97 
98  mLogger->trace("Add observer PLUGINNAME = %\n", mSamReaderPlugin->getName());
99 
100  std::dynamic_pointer_cast<ObservablePlugin>(mSamReaderPlugin)->addObserver(pluginObserver);
101  }

119 std::shared_ptr<SeResource<CalypsoSam>>
120  SamResourceManagerDefault::allocateSamResource(
121  const AllocationMode allocationMode,
122  const std::shared_ptr<SamIdentifier> samIdentifier)
123 {
124  const uint64_t maxBlockingDate = System::currentTimeMillis() + mMaxBlockingTime;
125  bool noSamResourceLogged = false;
126 
127  mLogger->trace("Allocating SAM reader channel...\n");
128 
129  while (true) {
130  mMutex.lock();
131  for (const auto& entry : mLocalManagedSamResources) {
132  std::shared_ptr<ManagedSamResource> managedSamResource = entry.second;
133  if (managedSamResource->isSamResourceFree()) {
134  if (managedSamResource->isSamMatching(samIdentifier)) {
135  managedSamResource->setSamResourceStatus(SamResourceStatus::BUSY);
136  mLogger->debug("Allocation succeeded. SAM resource created\n");
137  mMutex.unlock();
138  return managedSamResource;
139  }
140  }
141  }
142  mMutex.unlock();
143 
144  /*
145  * Loop until MAX_BLOCKING_TIME in blocking mode, only once in
146  * non-blocking mode
147  */
148  if (allocationMode == AllocationMode::NON_BLOCKING) {
149  mLogger->trace("No SAM resources available at the moment\n");
150  throw CalypsoNoSamResourceAvailableException(
151  "No Sam resource could be allocated for samIdentifier " +
152  samIdentifier->getGroupReference());
153  } else {
154  if (!noSamResourceLogged) {
155  /* Log once the first time */
156  mLogger->trace("No SAM resources available at the moment\n");
157  noSamResourceLogged = true;
158  }
159 
160  try {
161  Thread::sleep(mSleepTime);
162  } catch (const InterruptedException& e) {
163  /* Set interrupt flag */
164  /*
165  * C++ vs. Java: cannot interrupt thread as easily as in Java,
166  * as this scenario is quite unlikely to happen,
167  * will throw instead for now.
168  */
169  (void)e;
170  //Thread::currentThread()->interrupt();
171  mLogger->error("Interrupt exception in Thread::sleep\n");
172  throw;
173  }
174 
175  if (System::currentTimeMillis() >=
176  static_cast<unsigned long long>(maxBlockingDate)) {
177  mLogger->error(
178  "The allocation process failed. Timeout % sec exceeded\n",
179  (mMaxBlockingTime / 1000.0));
180 
181  std::stringstream ss;
182  ss << "No Sam resource could be allocated within"
183  << "timeout of " << mMaxBlockingTime << "ms "
184  << "for samIdentifier "<< samIdentifier->getGroupReference();
185 
186  throw CalypsoNoSamResourceAvailableException(ss.str());
187  }
188  }
189  }

208 void SamResourceManagerDefault::initSamReader(
209  const std::shared_ptr<SeReader> samReader,
210  const std::shared_ptr<ReaderObserver> readerObserver)
211 {
212  samReader->addSeProtocolSetting(SeCommonProtocols::PROTOCOL_ISO7816_3,".*");
213 
214  /* Specific to PCSC reader (no effect on Stub) */
215 
216  try {
217  /* Contactless SE works with T0 protocol */
218  samReader->setParameter("protocol", "T0");
219 
220  /* Shared mode */
221  samReader->setParameter("mode", "shared");
222  if (samReader->isSePresent()) {
223  mLogger->trace("Create SAM resource: %\n", samReader->getName());
224 
225  mMutex.lock();
226  mLocalManagedSamResources.insert({samReader->getName(),
227  createSamResource(samReader)});
228  mMutex.unlock();
229  }
230  } catch (const KeypleException& e) {
231  (void)e;
232  throw IllegalArgumentException("Parameters are not supported for this" \
233  " reader : protocol:TO, mode:shared");
234  }
235 
236  std::shared_ptr<ObservableReader> obsReader =
237  std::dynamic_pointer_cast<ObservableReader>(samReader);
238 
239  if (obsReader && readerObserver != nullptr) {
240  mLogger->trace("Add observer and start detection READERNAME = %\n",
241  samReader->getName());
242 
243  obsReader->addObserver(readerObserver);
244  obsReader->startSeDetection(PollingMode::REPEATING);
245  } else {
246  mLogger->trace("Sam Reader is not an ObservableReader = %\n",
247  samReader->getName());
248  }

261 void PluginObserver::update(const std::shared_ptr<PluginEvent> event)
262 {
263  for (const auto& readerName : event->getReaderNames()) {
264  std::shared_ptr<SeReader> samReader = nullptr;
265 
266  mLogger->info("PluginEvent: PLUGINNAME = %, READERNAME = %, EVENTTYPE" \
267  " = %\n",
268  event->getPluginName(),
269  readerName,
270  event->getEventType());
271 
272  /* We retrieve the reader object from its name. */
273  try {
274  samReader = SeProxyService::getInstance()
275  .getPlugin(event->getPluginName())
276  ->getReader(readerName);
277  } catch (const KeyplePluginNotFoundException& e) {
278  (void)e;
279  mLogger->error("Plugin not found %\n", event->getPluginName());
280  return;
281  } catch (const KeypleReaderNotFoundException& e) {
282  (void)e;
283  mLogger->error("Reader not found %\n", readerName);
284  return;
285  }
286 
287  const PluginEvent::EventType& type = event->getEventType();
288  if (type == PluginEvent::EventType::READER_CONNECTED) {
289  if (mParent.mLocalManagedSamResources.find(readerName) !=
290  mParent.mLocalManagedSamResources.end()) {
291  mLogger->trace("Reader is already present in the local " \
292  "samResources - READERNAME = %\n",
293  readerName);
294 
295  /* Do nothing */
296  return;
297  }
298 
299  mLogger->trace("New reader! READERNAME = %\n", samReader->getName());
300 
301  /*
302  * We are informed here of a connection of a reader.
303  *
304  * We add an observer to this reader if possible.
305  */
306  mPattern = Pattern::compile(mSamReaderFilter);
307  if (mPattern->matcher(readerName)->matches()) {
308  /* Enable logging */
309  try {
310  mParent.initSamReader(samReader, mReaderObserver);
311  } catch (const KeypleReaderException& e) {
312  mLogger->error("Unable to init Sam reader %\n",
313  samReader->getName(),
314  e.getCause().what());
315  }
316  } else {
317  mLogger->trace("Reader not matching: %\n", readerName);
318  }
319 
320  } else if (type == PluginEvent::EventType::READER_DISCONNECTED) {
321  /*
322  * We are informed here of a disconnection of a reader.
323  *
324  * The reader object still exists but will be removed from the
325  * reader list right after. Thus, we can properly remove the
326  * observer attached to this reader before the list update.
327  */
328  mPattern = Pattern::compile(mSamReaderFilter);
329  if (mPattern->matcher(readerName)->matches()) {
330  mLogger->trace("Reader removed. READERNAME = %\n", readerName);
331 
332  std::shared_ptr<ObservableReader> obsReader =
333  std::dynamic_pointer_cast<ObservableReader>(samReader);
334  if (obsReader) {
335  if (mReaderObserver != nullptr) {
336  mLogger->trace("Remove observer and stop detection " \
337  "READERNAME = %\n",
338  readerName);
339  obsReader->removeObserver(mReaderObserver);
340  obsReader->stopSeDetection();
341  } else {
342  mParent.removeResource(samReader);
343  mLogger->trace("Unplugged reader READERNAME = % " \
344  "wasn't observed. Resource removed\n",
345  readerName);
346  }
347  }
348  } else {
349  mLogger->trace("Reader not matching: %\n", readerName);
350  }
351 
352  } else {
353  mLogger->warn("Unexpected reader event-> EVENT = %\n",
354  event->getEventType().getName());
355  }
356  }
357 }

364 void ReaderObserver::update(const std::shared_ptr<ReaderEvent> event)
365 {
366  /* TODO revise exception management */
367  std::shared_ptr<SeReader> samReader = nullptr;
368 
369  try {
370  samReader = mParent.mSamReaderPlugin->getReader(event->getReaderName());
371  } catch (const KeypleReaderNotFoundException& e) {
372  mLogger->error("KeypleReaderNotFoundException raised, %\n", e);
373  }
374 
375  const std::lock_guard<std::mutex> lock(mParent.mMutex);
376 
377  const ReaderEvent::EventType& type = event->getEventType();
378  if (type == ReaderEvent::EventType::SE_MATCHED ||
379  type == ReaderEvent::EventType::SE_INSERTED) {
380  if (mParent.mLocalManagedSamResources.find(samReader->getName()) !=
381  mParent.mLocalManagedSamResources.end()) {
382  mLogger->trace("Reader is already present in the local " \
383  "samResources - READERNAME = %\n",
384  samReader->getName());
385 
386  /* Do nothing */
387  return;
388  }
389 
390  std::shared_ptr<ManagedSamResource> newSamResource = nullptr;
391 
392  try {
393  /*
394  * Although the reader allocation is dynamic, the SAM resource type
395  * is STATIC
396  */
397  newSamResource = mParent.createSamResource(samReader);
398  } catch (const CalypsoNoSamResourceAvailableException& e) {
399  (void)e;
400  mLogger->error("Failed to create a SeResource<CalypsoSam> from %\n",
401  samReader->getName());
402  }
403 
404  /* Failures are ignored */
405  if (newSamResource != nullptr) {
406  mLogger->trace("Created SAM resource: READER = %, SAM_REVISION = " \
407  "%, SAM_SERIAL_NUMBER = %\n",
408  event->getReaderName(),
409  newSamResource->getMatchingSe()->getSamRevision(),
410  newSamResource->getMatchingSe()->getSerialNumber());
411 
412  mParent.mLocalManagedSamResources.insert({samReader->getName(),
413  newSamResource});
414  }
415 
416  } else if (type == ReaderEvent::EventType::SE_REMOVED ||
417  type == ReaderEvent::EventType::TIMEOUT_ERROR) {
418  mParent.removeResource(samReader);
419  }
420 }
421 
422 }
