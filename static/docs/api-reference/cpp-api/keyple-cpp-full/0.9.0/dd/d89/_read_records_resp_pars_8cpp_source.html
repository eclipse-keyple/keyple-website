108 std::map<int, std::vector<uint8_t>> ReadRecordsRespPars::getRecords() const
109 {
110  std::map<int, std::vector<uint8_t>> records;
111 
112  if ((dynamic_cast<ReadRecordsCmdBuild*>(mBuilder))->getReadMode() ==
113  ReadRecordsCmdBuild::ReadMode::ONE_RECORD) {
114  records.insert({
115  (dynamic_cast<ReadRecordsCmdBuild*>(mBuilder))
116  ->getFirstRecordNumber(),
117  mResponse->getDataOut()});
118  } else {
119  std::vector<uint8_t> apdu = mResponse->getDataOut();
120  int apduLen = apdu.size();
121  int index = 0;
122  while (apduLen > 0) {
123  char recordNb = apdu[index++];
124  char len = apdu[index++];
125  records.insert(std::pair<int, std::vector<uint8_t>>(
126  recordNb, Arrays::copyOfRange(apdu, index, index + len)));
127  index = index + len;
128  apduLen = apduLen - 2 - len;
129  }
130  }
131 
132  return records;
133 }
