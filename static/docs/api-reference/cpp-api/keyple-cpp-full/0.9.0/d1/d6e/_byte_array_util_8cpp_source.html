53 std::vector<uint8_t> ByteArrayUtil::fromHex(const std::string& hex)
54 {
55  std::string _hex = HEX_IGNORED_CHARS->matcher(hex)->replaceAll("");
56  for (auto& c : _hex)
57  c = static_cast<char>(toupper((unsigned char)c));
58 
59  if (_hex.length() % 2 != 0)
60  throw IllegalArgumentException("Odd numbered hex array");
61 
62  std::vector<uint8_t> byteArray(_hex.length() / 2);
63  for (int i = 0; i < (int)_hex.length(); i += 2) {
64  byteArray[i / 2] = (Character::digit(_hex[i], 16) << 4) + Character::digit(_hex[i + 1], 16);
65  }
66 
67  return byteArray;
68 }
