42 static const std::string ATR1 = "3B001122805A0180D002030411223344829000";
43 /* The default KIF values for personalization, loading and debiting */
44 static const uint8_t DEFAULT_KIF_PERSO = 0x21;
45 static const uint8_t DEFAULT_KIF_LOAD = 0x27;
46 static const uint8_t DEFAULT_KIF_DEBIT = 0x30;
47 static const uint8_t DEFAULT_KVC_PERSO = 0x11;
48 static const uint8_t DEFAULT_KVC_LOAD = 0x22;
49 static const uint8_t DEFAULT_KVC_DEBIT = 0x33;
50 /*
51  * The default key record number values for personalization, loading and debiting
52  * The actual value should be adjusted.
53  */
54 static const uint8_t DEFAULT_KEY_RECORD_NUMBER_PERSO = 0x01;
55 static const uint8_t DEFAULT_KEY_RECORD_NUMBER_LOAD = 0x02;
56 static const uint8_t DEFAULT_KEY_RECORD_NUMBER_DEBIT = 0x03;

58 static std::shared_ptr<CalypsoSam> createCalypsoSam()
59 {
60  auto atr = std::make_shared<AnswerToReset>(ByteArrayUtil::fromHex(ATR1));
61  auto selectionStatus = std::make_shared<SelectionStatus>(atr, nullptr, true);
62  auto response = std::make_shared<SeResponse>(true, true, selectionStatus, nullptr);
63  return std::make_shared<CalypsoSam>(response, TransmissionMode::CONTACTS);
64 }

66 TEST(PoSecuritySettingsTest, poSecuritySettings_nominal)
67 {
68  std::shared_ptr<SeReader> samReader = nullptr;
69  std::shared_ptr<CalypsoSam> calypsoSam = createCalypsoSam();
70  auto samResource = std::make_shared<SeResource<CalypsoSam>>(samReader, calypsoSam);
71  auto poSecuritySettings =
72  PoSecuritySettingsBuilder(samResource)
73  .sessionDefaultKif(AccessLevel::SESSION_LVL_PERSO, DEFAULT_KIF_PERSO)
74  .sessionDefaultKif(AccessLevel::SESSION_LVL_LOAD, DEFAULT_KIF_LOAD)
75  .sessionDefaultKif(AccessLevel::SESSION_LVL_DEBIT, DEFAULT_KIF_DEBIT)
76  .sessionDefaultKvc(AccessLevel::SESSION_LVL_PERSO, DEFAULT_KVC_PERSO)
77  .sessionDefaultKvc(AccessLevel::SESSION_LVL_LOAD, DEFAULT_KVC_LOAD)
78  .sessionDefaultKvc(AccessLevel::SESSION_LVL_DEBIT, DEFAULT_KVC_DEBIT)
79  .sessionDefaultKeyRecordNumber(AccessLevel::SESSION_LVL_PERSO,
80  DEFAULT_KEY_RECORD_NUMBER_PERSO)
81  .sessionDefaultKeyRecordNumber(AccessLevel::SESSION_LVL_LOAD,
82  DEFAULT_KEY_RECORD_NUMBER_LOAD)
83  .sessionDefaultKeyRecordNumber(AccessLevel::SESSION_LVL_DEBIT,
84  DEFAULT_KEY_RECORD_NUMBER_DEBIT)
85  .build();
86 
87  ASSERT_EQ(poSecuritySettings->getSessionDefaultKif(AccessLevel::SESSION_LVL_PERSO),
88  DEFAULT_KIF_PERSO);
89  ASSERT_EQ(poSecuritySettings->getSessionDefaultKif(AccessLevel::SESSION_LVL_LOAD),
90  DEFAULT_KIF_LOAD);
91  ASSERT_EQ(poSecuritySettings->getSessionDefaultKif(AccessLevel::SESSION_LVL_DEBIT),
92  DEFAULT_KIF_DEBIT);
93  ASSERT_EQ(poSecuritySettings->getSessionDefaultKvc(AccessLevel::SESSION_LVL_PERSO),
94  DEFAULT_KVC_PERSO);
95  ASSERT_EQ(poSecuritySettings->getSessionDefaultKvc(AccessLevel::SESSION_LVL_LOAD),
96  DEFAULT_KVC_LOAD);
97  ASSERT_EQ(poSecuritySettings->getSessionDefaultKvc(AccessLevel::SESSION_LVL_DEBIT),
98  DEFAULT_KVC_DEBIT);
99  ASSERT_EQ(poSecuritySettings->getSessionDefaultKeyRecordNumber(AccessLevel::SESSION_LVL_PERSO),
100  DEFAULT_KEY_RECORD_NUMBER_PERSO);
101  ASSERT_EQ(poSecuritySettings->getSessionDefaultKeyRecordNumber(AccessLevel::SESSION_LVL_LOAD),
102  DEFAULT_KEY_RECORD_NUMBER_LOAD);
103  ASSERT_EQ(poSecuritySettings->getSessionDefaultKeyRecordNumber(AccessLevel::SESSION_LVL_DEBIT),
104  DEFAULT_KEY_RECORD_NUMBER_DEBIT);
105  ASSERT_EQ(poSecuritySettings->getSamResource(), samResource);
106 }
