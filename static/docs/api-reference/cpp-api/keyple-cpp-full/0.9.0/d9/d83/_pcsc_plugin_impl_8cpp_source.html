63 const std::set<std::string>& PcscPluginImpl::fetchNativeReadersNames()
64 {
65  nativeReadersNames.clear();
66 
67  try {
68  const std::vector<PcscTerminal>& terminals = getTerminals();
69 
70  logger->debug("fetchNativeReadersNames - filling native readers name "
71  "list with card terminal items\n");
72 
73  if (terminals.empty()) {
74  logger->trace("fetchNativeReadersNames - no reader available\n");
75  } else {
76  for (auto& term : terminals) {

86  throw KeypleReaderIOException("Could not access terminals list", e);
87  }

92 ConcurrentMap<const std::string, std::shared_ptr<SeReader>>& PcscPluginImpl::initNativeReaders()
93 {
94  /* C++ vs. Java: clear class member instead of creating new map */
95  mNativeReaders.clear();
96 
97  /*
98  * activate a special processing "SCARD_E_NO_NO_SERVICE" (on Windows
99  * platforms the removal of the last PC/SC reader stops the "Windows Smart
100  * Card service")
101  */
102 #ifdef _WIN32
103  scardNoServiceHackNeeded = true;
104  logger->info("System detected : %\n", scardNoServiceHackNeeded);
105 #else
106  scardNoServiceHackNeeded = false;
107 #endif
108 
109  /*
110  * Parse the current readers list to create the ProxyReader(s) associated
111  * with new reader(s).
112  */
113  logger->debug("initNativeReaders - getting card terminals\n");
114  std::vector<PcscTerminal>& terminals = getTerminals();
115  logger->trace("[%] initNativeReaders => CardTerminal in list: %\n", getName(), terminals);
116 
117  if (terminals.empty()) {
118  logger->trace("No reader available\n");
119  return mNativeReaders;
120  }
121 
122  try {
123  for (auto& term : terminals) {
124  std::shared_ptr<PcscReaderImpl> pcscReader =
125  std::make_shared<PcscReaderImpl>(getName(), term);
126  mNativeReaders.insert({pcscReader->getName(), pcscReader});
127  }
128  } catch (PcscTerminalException& e) {
129  logger->trace("[%] terminal list not accessible, msg: %, cause: %",
130  getName(),
131  e.getMessage(),
132  e.getCause().what());
133  /*
134  * Throw new KeypleReaderIOException("Could not access terminals list",
135  * e); do not propagate exception at the constructor will propagate it
136  * as a keyple::core::seproxy::exception::KeypleRuntimeException/
137  */
138  }
139 
140  return mNativeReaders;
141 }

143 std::shared_ptr<SeReader> PcscPluginImpl::fetchNativeReader(const std::string& name)
144 {
145  /* Return the current reader if it is already listed */
146  ConcurrentMap<const std::string, std::shared_ptr<SeReader>> ::const_iterator it;
147  if ((it = mNativeReaders.find(name)) != mNativeReaders.end())
148  return it->second;
149 
150  /*
151  * Parse the current PC/SC readers list to create the ProxyReader(s)
152  * associated with new reader(s)
153  */
154  std::shared_ptr<SeReader> seReader = nullptr;
155  std::vector<PcscTerminal>& terminals = getTerminals();
156  //std::vector<std::string> terminalList;
157 
158  try {
159  for (auto& term : terminals) {
160  if (!term.getName().compare(name)) {
161  logger->trace("[%s] fetchNativeReader => CardTerminal in new PcscReader: %\n",
162  getName(),
163  terminals);
164  seReader = std::make_shared<PcscReaderImpl>(getName(), term);
165  }
166  }
167  } catch (const PcscTerminalException& e) {
168  logger->trace("[%] caught PcscTerminalException (msg: %, cause: %)\n",
169  getName(),
170  e.getMessage(),
171  e.getCause().what());
172  throw KeypleReaderIOException("Could not access terminals list", e);
173  }
174 
175  if (seReader == nullptr)
176  throw KeypleReaderNotFoundException("Reader " + name + " not found!");
177 
178  return seReader;
179 }
