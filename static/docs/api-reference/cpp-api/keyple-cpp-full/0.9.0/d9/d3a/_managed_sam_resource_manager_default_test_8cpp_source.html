164  const std::string& samFilter)
165 {
166  std::vector<std::shared_ptr<SeResponse>> selectionResponses;
167  selectionResponses.push_back(samSelectionSuccess());
168 
169  /* Create a mock reader */

172  EXPECT_CALL(*reader, getName()).WillRepeatedly(ReturnRef(SAM_READER_NAME));
173  EXPECT_CALL(*reader, isSePresent()).WillRepeatedly(Return(true));
174  EXPECT_CALL(*reader, transmitSeRequests(_,_,_)).WillRepeatedly(Return(selectionResponses));
175  if (samFilter == ".*") {
176  EXPECT_CALL(*reader, addSeProtocolSetting(_,_)).WillRepeatedly(Return());
177  EXPECT_CALL(*reader, setParameter(_,_)).WillRepeatedly(Return());
178  }
179 
180  /* Create a list of mock readers */
181  ConcurrentMap<const std::string, std::shared_ptr<SeReader>> readers;
182  readers.insert({reader->getName(), reader});
183 
184  /* Create the mock plugin */
185  auto plugin = std::make_shared<MSRMD_ReaderPluginMock>();
186 
187  EXPECT_CALL(*plugin, getReaders()).WillRepeatedly(ReturnRef(readers));
188  EXPECT_CALL(*plugin, getReader(SAM_READER_NAME)).WillRepeatedly(Return(reader));
189  EXPECT_CALL(*plugin, getName()).WillRepeatedly(ReturnRef(SAM_PLUGIN_NAME));
190 
191  return std::make_shared<SamResourceManagerDefault>(plugin, samFilter, 1000, 10);
192 }
