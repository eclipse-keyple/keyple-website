152 bool PcscReaderImpl::waitForCardPresent()
153 {
154  if (mShuttingDown)
155  return false;
156 
157  mLoopWaitSe = true;
158 
159  mLogger->trace("[%] waitForCardPresent\n", getName());
160 
161  try {
162 
163  while (mLoopWaitSe) {
164  if (mTerminal.isCardPresent(false)) {
165  /* Card inserted */
166  mLogger->trace("[%] card present\n", getName());
167  return true;
168  }
169 
170  try {
171  Thread::sleep(50);
172  } catch (const InterruptedException& e) {
173  mLogger->debug("Sleep was interrupted - %\n", e);
174  }
175  }
176 
177  mLogger->trace("[%] mLoopWaitSe=false, leaving\n", getName());
178 
179  /* If loop was stopped */
180  return false;
181 
182  } catch (PcscTerminalException& e) {
183  throw KeypleReaderIOException(
184  "[" + getName() + "] Exception occurred in " \
185  "waitForCardPresent. Message: " + e.getMessage());
186  }
187 }

249 bool PcscReaderImpl::protocolFlagMatches(const std::shared_ptr<SeProtocol> protocolFlag)
250 {
251  bool result;
252 
253  /* Test protocolFlag to check if ATR based protocol filtering is required */
254  if (protocolFlag) {
255  if (!isPhysicalChannelOpen()) {
256  mLogger->debug("protocolFlagMatches - physical channel not open, "
257  "opening it\n");
258  openPhysicalChannel();
259  }
260 
261  /* The request will be executed only if the protocol match the requestElement */
262  const auto& it = getProtocolsMap().find(protocolFlag);
263  if (it == getProtocolsMap().end())
264  throw KeypleReaderIOException("Target selector mask not found!");
265 
266  const std::string selectionMask = it->second;
267 
268  std::unique_ptr<Pattern> p = Pattern::compile(selectionMask);
269  std::string atr = ByteArrayUtil::toHex(mTerminal.getATR());
270  if (!p->matcher(atr)->matches()) {
271  mLogger->debug("[%] protocolFlagMatches => unmatching SE. "
272  "PROTOCOLFLAG = %\n", getName(), protocolFlag);
273  result = false;
274  } else {
275  mLogger->debug("[%] protocolFlagMatches => matching SE. "
276  "PROTOCOLFLAG = %\n", getName(), protocolFlag);
277  result = true;
278  }
279  } else {
280  /* No protocol defined returns true */
281  result = true;
282  }
283 
284  return result;
285 }

287 void PcscReaderImpl::setParameter(const std::string& name,
288  const std::string& value)
289 { mLogger->debug("[%] setParameter => PCSC: Set a parameter. NAME = %,"
290  " VALUE = %\n", getName(), name, value);
291 
292  if (name == "") {
293  throw std::invalid_argument("Parameter shouldn't be null\n");
294  }
295 
296  /*
297  * /!\ Must use hardcoded strings here as
298  * PcscReader::SETTING_KEY_TRANSMISSION_MODE and the three other strings
299  * are not yet instantiated.
300  */
301  if (name == PcscReader::SETTING_KEY_TRANSMISSION_MODE) {
302  if (value == "")
303  mTransmissionMode = TransmissionMode::NONE;
304  else if (value == "contacts")
305  mTransmissionMode = TransmissionMode::CONTACTS;
306  else if (value == "contactless")
307  mTransmissionMode = TransmissionMode::CONTACTLESS;
308  else
309  throw IllegalArgumentException(
310  "Bad tranmission mode " + name + " : " + value);
311 
312  } else if (name == PcscReader::SETTING_KEY_PROTOCOL) {
313 
314  /* /!\ C++ vs. Java - Clear parameters table */
315  mParameters.clear();
316 
317  if (value == "" || value == "Tx")
318  mParameterCardProtocol = "*";
319  else if (value == PcscReader::SETTING_PROTOCOL_T0)
320  mParameterCardProtocol = "T=0";
321  else if (value == PcscReader::SETTING_PROTOCOL_T1)
322  mParameterCardProtocol = "T=1";
323  else if (value == PcscReader::SETTING_PROTOCOL_T_CL)
324  mParameterCardProtocol = "T=CL";
325  else
326  throw IllegalArgumentException(
327  "Bad protocol " + name + " : " + value);
328 
329  /* /!\ C++ vs. Java */
330  mParameters.emplace(SETTING_KEY_PROTOCOL, mParameterCardProtocol);
331 
332  /*
333  * /!\ C++ vs. Java
334  *
335  * Actualize 'transmissionMode' according to 'parameterCardProtocol'.
336  */
337  if (mParameterCardProtocol == PROTOCOL_T1 ||
338  mParameterCardProtocol == PROTOCOL_T_CL) {
339  mTransmissionMode = TransmissionMode::CONTACTLESS;
340  } else {
341  mTransmissionMode = TransmissionMode::CONTACTS;
342  }
343 
344  } else if (name == PcscReader::SETTING_KEY_MODE) {
345  if (value == "" || value == PcscReader::SETTING_MODE_SHARED) {
346  if (mCardExclusiveMode) {
347  try {
348  mTerminal.endExclusive();
349  } catch (PcscTerminalException& e) {
350  throw KeypleReaderIOException("Couldn't disable exclusive "
351  "mode", e);
352  }
353  }
354  mCardExclusiveMode = false;
355 
356  /* /!\ C++ vs. Java */
357  mParameters.emplace(SETTING_KEY_MODE, SETTING_MODE_SHARED);
358 
359  } else if (value == PcscReader::SETTING_MODE_EXCLUSIVE) {
360  mCardExclusiveMode = true;
361  } else {
362  throw IllegalArgumentException(
363  "Parameter value not supported " + name + " : " + value);
364  }
365 
366 
367  } else if (name == PcscReader::SETTING_KEY_DISCONNECT) {
368  if (value == "" || value == PcscReader::SETTING_DISCONNECT_RESET) {
369  mCardReset = true;
370  } else if (value == PcscReader::SETTING_DISCONNECT_UNPOWER) {
371  mCardReset = false;
372  } else if (value == PcscReader::SETTING_DISCONNECT_EJECT ||
373  value == PcscReader::SETTING_DISCONNECT_LEAVE) {
374  throw IllegalArgumentException(
375  "This disconnection parameter is not supported by this " \
376  "plugin" + name + " : " + value);
377  } else {
378  throw IllegalArgumentException(
379  "Parameters not supported : " + name + " : " + value);
380  }
381  } else {
382  throw IllegalArgumentException(
383  "This parameter is unknown! " + name + " : " + value);
384  }
385 }
