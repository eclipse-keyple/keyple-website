40 void PoCommandManager::addStoredValueCommand(
41  std::shared_ptr<AbstractPoCommandBuilder<AbstractPoResponseParser>> commandBuilder,
42  const Operation& svOperation)
43 {
44  /* Check the logic of the SV command sequencing */
45  auto calypso = std::dynamic_pointer_cast<CalypsoPoCommand>(commandBuilder->getCommandRef());
46  CalypsoPoCommand command = *calypso;
47  if (command == CalypsoPoCommand::SV_GET) {
48  mSvOperation = svOperation;
49  } else if (command == CalypsoPoCommand::SV_RELOAD ||
50  command == CalypsoPoCommand::SV_DEBIT ||
51  command == CalypsoPoCommand::SV_UNDEBIT) {
52  if (!mPoCommands.empty())
53  throw CalypsoPoTransactionIllegalStateException("This SV command can only be placed " \
54  "in the first position in the list of" \
55  " prepared commands");
56 
57  if (*mSvLastCommand != CalypsoPoCommand::SV_GET)
58  /* @see Calypso Layer ID 8.07/8.08 (200108) */
59  throw IllegalStateException("This SV command must follow an SV Get command");
60 
61  /* Here, we expect the builder and the SV operation to be consistent */
62  if (svOperation != mSvOperation) {
63  mLogger->error("Sv operation = %, current command = %", mSvOperation, svOperation);
64  throw CalypsoPoTransactionIllegalStateException("Inconsistent SV operation.");
65  }
66 
67  mSvOperation = svOperation;
68  mSvOperationComplete = true;
69  } else {
70  throw IllegalStateException("An SV command is expected.");
71  }
72 
73  mSvLastCommand = calypso;
74 
75  mPoCommands.push_back(commandBuilder);
76 }
