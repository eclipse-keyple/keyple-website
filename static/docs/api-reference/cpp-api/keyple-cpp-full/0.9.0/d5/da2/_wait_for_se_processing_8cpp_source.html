33 void WaitForSeProcessing::onEvent(const InternalEvent event)
34 {
35  mLogger->trace("[%] onEvent => Event % received in currentState %\n",
36  mReader->getName(),
37  event,
38  mState);
39 
40  /* Process InternalEvent */
41  switch (event) {
42  case InternalEvent::SE_PROCESSED:
43  if (this->mReader->getPollingMode() == ObservableReader::PollingMode::REPEATING) {
44  switchState(MonitoringState::WAIT_FOR_SE_REMOVAL);
45  } else {
46  /* We close the channels now and notify the application of the SE_REMOVED event */
47  this->mReader->processSeRemoved();
48  switchState(MonitoringState::WAIT_FOR_START_DETECTION);
49  }
50  break;
51  case InternalEvent::SE_REMOVED:
52  /*
53  * The SE has been removed, we close all channels and return to
54  * the currentState of waiting for insertion
55  * We notify the application of the SE_REMOVED event.
56  */
57  mReader->processSeRemoved();
58  if (mReader->getPollingMode() == ObservableReader::PollingMode::REPEATING) {
59  switchState(MonitoringState::WAIT_FOR_SE_INSERTION);
60  } else {
61  switchState(MonitoringState::WAIT_FOR_START_DETECTION);
62  }
63  break;
64  case InternalEvent::STOP_DETECT:
65  mReader->processSeRemoved();
66  switchState(MonitoringState::WAIT_FOR_START_DETECTION);
67  break;
68 
69  default:
70  mLogger->warn("[%] Ignore => Event % received in currentState %\n",
71  mReader->getName(),
72  event,
73  mState);
74  break;
75  }
