88 const std::vector<uint8_t> SamCommandProcessor::getSessionTerminalChallenge()
89 {
90  std::vector<std::shared_ptr<ApduRequest>> apduRequests;
91 
92  /* Diversify only if this has not already been done. */
93  if (!mIsDiversificationDone) {
94  /*
95  * Build the SAM Select Diversifier command to provide the SAM with the
96  * PO S/N
97  */
98  std::shared_ptr<AbstractApduCommandBuilder> selectDiversifier =
99  std::make_shared<SelectDiversifierCmdBuild>(
100  mSamResource->getMatchingSe()->getSamRevision(),
101  mPoResource->getMatchingSe()->getApplicationSerialNumberBytes());
102 
103  apduRequests.push_back(selectDiversifier->getApduRequest());
104 
105  /* Note that the diversification has been made */
106  mIsDiversificationDone = true;
107  }
108 
109  /* Build the SAM Get Challenge command */
110  const uint8_t challengeLength =
111  mPoResource->getMatchingSe()->isConfidentialSessionModeSupported() ?
112  CHALLENGE_LENGTH_REV32 : CHALLENGE_LENGTH_REV_INF_32;
113 
114  std::shared_ptr<AbstractSamCommandResponse>
115  getChallengeCmdBuild =
116  std::reinterpret_pointer_cast<
117  AbstractSamCommandResponse>(
118  std::make_shared<SamGetChallengeCmdBuild>(
119  mSamResource->getMatchingSe()->getSamRevision(),
120  challengeLength));
121 
122  apduRequests.push_back(getChallengeCmdBuild->getApduRequest());
123 
124  /*
125  * Transmit the SeRequest to the SAM and get back the SeResponse (list of
126  * ApduResponse)
127  */
128  std::shared_ptr<SeResponse> samSeResponse;
129 
130  try {
131  samSeResponse =
132  mSamReader->transmitSeRequest(
133  std::make_shared<SeRequest>(apduRequests),
134  ChannelControl::KEEP_OPEN);
135  } catch (const KeypleReaderIOException& e) {
136  throw CalypsoSamIOException(
137  "SAM IO Exception while getting terminal challenge.", e);
138  }
139 
140  const std::vector<std::shared_ptr<ApduResponse>> samApduResponses =
141  samSeResponse->getApduResponses();
142 
143  std::vector<uint8_t> sessionTerminalChallenge;
144 
145  const int numberOfSamCmd = apduRequests.size();
146  if (static_cast<int>(samApduResponses.size()) == numberOfSamCmd) {
147  auto getChallengeRespPars =
148  std::dynamic_pointer_cast<SamGetChallengeRespPars>(
149  getChallengeCmdBuild->createResponseParser(
150  samApduResponses[numberOfSamCmd - 1]));
151 
152  getChallengeRespPars->checkStatus();
153 
154  sessionTerminalChallenge = getChallengeRespPars->getChallenge();
155 
156  mLogger->debug("identification: TERMINALCHALLENGE = %\n",
157  sessionTerminalChallenge);
158 
159  } else {
160  std::stringstream ss;
161  ss << "The number of commands/responses does not "
162  << "match: cmd=" << numberOfSamCmd << ", resp="
163  << samApduResponses.size();
164  throw CalypsoDesynchronizedExchangesException(ss.str());
165  }
166 
167  return sessionTerminalChallenge;
168 }

223 void SamCommandProcessor::pushPoExchangeData(
224  const std::shared_ptr<ApduRequest> request,
225  const std::shared_ptr<ApduResponse> response)
226 {
227  mLogger->trace("pushPoExchangeData: REQUEST = %\n", request);
228 
229  /*
230  * Add an ApduRequest to the digest computation: if the request is of case4
231  * type, Le must be excluded from the digest computation. In this cas, we
232  * remove here the last byte of the command buffer.
233  */
234  if (request->isCase4())
235  mPoDigestDataCache.push_back(
236  Arrays::copyOfRange(request->getBytes(),
237  0,
238  request->getBytes().size() - 1));
239  else
240  mPoDigestDataCache.push_back(request->getBytes());
241 
242  mLogger->trace("pushPoExchangeData: RESPONSE = %\n", response);
243 
244  /* Add an ApduResponse to the digest computation */
245  mPoDigestDataCache.push_back(response->getBytes());
246 }

258 const std::vector<std::shared_ptr<AbstractSamCommandResponse>>
259  SamCommandProcessor::getPendingSamCommands(const bool addDigestClose)
260 {
261  /* TODO optimization with the use of Digest Update Multiple whenever possible */
262  std::vector<std::shared_ptr<AbstractSamCommandResponse>> samCommands;
263 
264  /* Sanity checks */
265  if (mPoDigestDataCache.empty()) {
266  mLogger->debug("getSamDigestRequest: no data in cache\n");
267  throw IllegalStateException("Digest data cache is empty.");
268  }
269 
270  if (!mIsDigestInitDone && mPoDigestDataCache.size() % 2 == 0) {
271  /* The number of buffers should be 2*n + 1 */
272  mLogger->debug("getSamDigestRequest: wrong number of buffer in cache NBR = %\n.",
273  mPoDigestDataCache.size());
274  throw IllegalStateException("Digest data cache is inconsistent.");
275  }
276 
277  if (!mIsDigestInitDone) {
278  /*
279  * Build and append Digest Init command as first ApduRequest of the digest computation
280  * process. The Digest Init command comes from the Open Secure Session response from the PO.
281  * Once added to the ApduRequest list, the data is remove from the cache to keep only
282  * couples of PO request/response
283  */
284  auto init = std::make_shared<DigestInitCmdBuild>(
285  mSamResource->getMatchingSe()->getSamRevision(),
286  mVerificationMode,
287  mPoResource->getMatchingSe()->isConfidentialSessionModeSupported(),
288  mWorkKeyRecordNumber,
289  mWorkKeyKif,
290  mWorkKeyKVC,
291  mPoDigestDataCache[0]);
292  auto cmd = std::reinterpret_pointer_cast<AbstractSamCommandResponse>(init);
293  samCommands.push_back(cmd);
294 
295  mPoDigestDataCache.erase(mPoDigestDataCache.begin());
296 
297  /* Note that the digest init has been made */
298  mIsDigestInitDone = true;
299  }
300 
301  /* Build and append Digest Update commands */
302  for (int i = 0; i < static_cast<int>(mPoDigestDataCache.size()); i++) {
303  auto update = std::make_shared<DigestUpdateCmdBuild>(
304  mSamResource->getMatchingSe()->getSamRevision(),
305  mSessionEncryption,
306  mPoDigestDataCache[i]);
307  auto cmd = std::reinterpret_pointer_cast<AbstractSamCommandResponse>(update);
308  samCommands.push_back(cmd);
309  }
310 
311  /* clears cached commands once they have been processed */
312  mPoDigestDataCache.clear();
313 
314  if (addDigestClose) {
315  /* Build and append Digest Close command */
316  auto close = std::make_shared<DigestCloseCmdBuild>(
317  mSamResource->getMatchingSe()->getSamRevision(),
318  mPoResource->getMatchingSe()->getRevision() == PoRevision::REV3_2 ?
319  SIGNATURE_LENGTH_REV32 : SIGNATURE_LENGTH_REV_INF_32);
320  auto cmd = std::reinterpret_pointer_cast<AbstractSamCommandResponse>(close);
321  samCommands.push_back(cmd);
322  }
323 
324  return samCommands;
325 }

327 const std::vector<uint8_t> SamCommandProcessor::getTerminalSignature()
328 {
329  /*
330  * All remaining SAM digest operations will now run at once.
331  * Get the SAM Digest request including Digest Close from the cache manager
332  */
333  std::vector<std::shared_ptr<AbstractSamCommandResponse>> samCommands =
334  getPendingSamCommands(true);
335 
336  auto samSeRequest = std::make_shared<SeRequest>(getApduRequests(samCommands));
337 
338  /* Transmit SeRequest and get SeResponse */
339  std::shared_ptr<SeResponse> samSeResponse;
340 
341  try {
342  samSeResponse = mSamReader->transmitSeRequest(samSeRequest, ChannelControl::KEEP_OPEN);
343  } catch (const KeypleReaderIOException& e) {
344  throw CalypsoSamIOException("SAM IO Exception while transmitting digest data.", e);
345  }
346 
347  const std::vector<std::shared_ptr<ApduResponse>>& samApduResponses =
348  samSeResponse->getApduResponses();
349 
350  if (samApduResponses.size() != samCommands.size()) {
351  std::stringstream ss;
352  ss << "The number of commands/responses does not match: cmd=" << samCommands.size()
353  << ", resp=" << samApduResponses.size();
354  throw CalypsoDesynchronizedExchangesException(ss.str());
355  }
356 
357  /* Check all responses status */
358  for (int i = 0; i < static_cast<int>(samApduResponses.size()); i++)
359  samCommands[i]->createResponseParser(samApduResponses[i])->checkStatus();
360 
361  /* Get Terminal Signature from the latest response */
362  std::shared_ptr<AbstractSamResponseParser> response =
363  samCommands[samCommands.size() - 1]->createResponseParser(
364  samApduResponses[samCommands.size() - 1]);
365  auto digestCloseRespPars = std::reinterpret_pointer_cast<DigestCloseRespPars>(response);
366 
367  const std::vector<uint8_t>& sessionTerminalSignature = digestCloseRespPars->getSignature();
368 
369  mLogger->debug("SIGNATURE = %\n", sessionTerminalSignature);
370 
371  return sessionTerminalSignature;
372 }

428 const std::vector<uint8_t> SamCommandProcessor::getCipheredPinData(
429  const std::vector<uint8_t>& poChallenge,
430  const std::vector<uint8_t>& currentPin,
431  const std::vector<uint8_t>& newPin)
432 {
433  std::vector<std::shared_ptr<AbstractSamCommandResponse>> samCommands;
434  std::shared_ptr<KeyReference> pinCipheringKeyRef;
435 
436  if (mWorkKeyKif != 0) {
437  /* The current work key has been set (a secure session is open) */
438  pinCipheringKeyRef = std::make_shared<KeyReference>(mWorkKeyKif, mWorkKeyKVC);
439  } else {
440  /* No current work key is available (outside secure session) */
441  pinCipheringKeyRef = mPoSecuritySettings->getDefaultPinCipheringKey();
442  }
443 
444  if (!mIsDiversificationDone) {
445  /* Build the SAM Select Diversifier command to provide the SAM with the PO S/N */
446  auto select = std::make_shared<SelectDiversifierCmdBuild>(
447  mSamResource->getMatchingSe()->getSamRevision(),
448  mPoResource->getMatchingSe()->getApplicationSerialNumberBytes());
449  auto cmd = std::reinterpret_pointer_cast<AbstractSamCommandResponse>(select);
450  samCommands.push_back(cmd);
451  mIsDiversificationDone = true;
452  }
453 
454  if (mIsDigesterInitialized) {
455  /* Get the pending SAM ApduRequest and add it to the current ApduRequest list */
456  const std::vector<std::shared_ptr<AbstractSamCommandResponse>> pending =
457  getPendingSamCommands(false);
458  samCommands.insert(std::end(samCommands), std::begin(pending), std::end(pending));
459  }
460 
461  auto random = std::make_shared<GiveRandomCmdBuild>(
462  mSamResource->getMatchingSe()->getSamRevision(),
463  poChallenge);
464  auto cmd = std::reinterpret_pointer_cast<AbstractSamCommandResponse>(random);
465  samCommands.push_back(cmd);
466 
467 
468  int cardCipherPinCmdIndex = samCommands.size();
469 
470  auto cardCipherPinCmdBuild = std::make_shared<CardCipherPinCmdBuild>(
471  mSamResource->getMatchingSe()->getSamRevision(),
472  pinCipheringKeyRef,
473  currentPin,
474  newPin);
475  auto cipher = std::make_shared<CardCipherPinCmdBuild>(
476  mSamResource->getMatchingSe()->getSamRevision(),
477  pinCipheringKeyRef,
478  currentPin,
479  newPin);
480  cmd = std::reinterpret_pointer_cast<AbstractSamCommandResponse>(cipher);
481  samCommands.push_back(cmd);
482 
483  /* Build a SAM SeRequest */
484  auto samSeRequest = std::make_shared<SeRequest>(getApduRequests(samCommands));
485 
486  /* Execute the command */
487  std::shared_ptr<SeResponse> samSeResponse =
488  mSamReader->transmitSeRequest(samSeRequest, ChannelControl::KEEP_OPEN);
489 
490  std::shared_ptr<ApduResponse> cardCipherPinResponse =
491  samSeResponse->getApduResponses()[cardCipherPinCmdIndex];
492 
493  /* Create a parser */
494  std::shared_ptr<CardCipherPinRespPars> cardCipherPinRespPars =
495  cardCipherPinCmdBuild->createResponseParser(cardCipherPinResponse);
496 
497  cardCipherPinRespPars->checkStatus();
498 
499  return cardCipherPinRespPars->getCipheredData();
500 }
