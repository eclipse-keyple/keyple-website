163 std::shared_ptr<AbstractOpenSessionRespPars>
164 AbstractOpenSessionRespPars::create(std::shared_ptr<ApduResponse> response,
165  PoRevision revision)
166 {
167  switch (revision) {
168  case PoRevision::REV1_0:
169  return std::make_shared<OpenSession10RespPars>(
170  response,
171  dynamic_cast<OpenSession10CmdBuild*>(mBuilder));
172  case PoRevision::REV2_4:
173  return std::make_shared<OpenSession24RespPars>(
174  response,
175  dynamic_cast<OpenSession24CmdBuild*>(mBuilder));
176  case PoRevision::REV3_1:
177  case PoRevision::REV3_1_CLAP:
178  return std::make_shared<OpenSession31RespPars>(
179  response,
180  dynamic_cast<OpenSession31CmdBuild*>(mBuilder));
181  case PoRevision::REV3_2:
182  return std::make_shared<OpenSession32RespPars>(
183  response,
184  dynamic_cast<OpenSession32CmdBuild*>(mBuilder));
185  default:
186  throw std::invalid_argument("Unknow revision " +
187  StringHelper::toString(revision));
188  }
189 }
